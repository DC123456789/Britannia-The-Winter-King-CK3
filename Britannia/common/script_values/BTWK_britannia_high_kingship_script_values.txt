BTWK_britannia_high_king_scope_support_for_claimant = {
	value = 0
	if = {
		limit = { is_ai = yes }
		
		# Opinion
		add = {
			value = "opinion(scope:claimant)"
			if = {
				limit = { "opinion(scope:claimant)" > 0 }
				divide = 2
			}
			else = {
				divide = 1.5
			}
			desc = CLAIMANT_SUPPORT_OPINION
		}
		# Hooks
		if = {
			limit = { scope:claimant = { has_strong_hook = prev } }
			add = {
				value = 25
				desc = CLAIMANT_SUPPORT_STRONG_HOOK
			}
		}
		else_if = {
			limit = { scope:claimant = { has_hook = prev } }
			add = {
				value = 10
				desc = CLAIMANT_SUPPORT_HOOK
			}
		}
		# Relative Power
		# 20 * (scope:claimant.military_power / military_power - 1.25 ) if the claimant is at least 1.25x stronger
		# -50 * (military_power / scope:claimant.military_power - 0.8) if the claimant is not at least 1.25x stronger
		add = {
			value = 0
			if = {
				limit = { BTWK_britannia_high_king_claimant_to_scope_military_power_ratio > 1.25 }
				add = {
					value = scope:claimant.military_power
					min = 1
				}
				divide = {
					value = military_power
					min = 1
				}
				subtract = 1.25
				multiply = 20
			}
			else = {
				add = {
					value = military_power
					min = 1
				}
				divide = {
					value = scope:claimant.military_power
					min = 1
				}
				subtract = 0.8
				multiply = -50
			}
			max = 50
			desc = CLAIMANT_SUPPORT_MILITARY_POWER
		}
		# Claimant prestige level
		add = {
			value = scope:claimant.prestige_level
			subtract = 3
			if = {
				limit = { scope:claimant.prestige_level < 3 }
				multiply = 10
			}
			else = {
				multiply = 20
			}
			min = 0
			desc = CLAIMANT_SUPPORT_PRESTIGE_LEVEL
		}
		# Claimant stats
		add = {
			value = scope:claimant.diplomacy
			subtract = 15
			if = {
				limit = { scope:claimant.diplomacy < 15 }
				multiply = 2
			}
			max = 15
			desc = CLAIMANT_SUPPORT_DIPLOMACY
		}
		add = {
			value = scope:claimant.martial
			subtract = 15
			if = {
				limit = { scope:claimant.martial < 15 }
				multiply = 2
			}
			max = 15
			desc = CLAIMANT_SUPPORT_MARTIAL
		}
		add = {
			value = scope:claimant.prowess
			subtract = 15
			if = {
				limit = { scope:claimant.prowess > 15 }
				multiply = 0.5
			}
			max = 10
			desc = CLAIMANT_SUPPORT_PROWESS
		}
		# Claimant Tribal Fame
		if = {
			limit = { scope:claimant = { government_has_flag = government_has_tribal_fame } }
			add = {
				value = scope:claimant.tribal_fame_value
				save_temporary_scope_value_as = claimant_tribal_fame_value
				if = {
					limit = { scope:claimant_tribal_fame_value > 0 }
					multiply = 0.5
				}
				desc = CLAIMANT_SUPPORT_TRIBAL_FAME
			}
		}
		# Claimant claims
		if = {
			limit = { scope:claimant = { has_strong_claim_on = title:e_britannia } }
			add = {
				value = 10
				desc = CLAIMANT_SUPPORT_PRESSED_CLAIM
			}
		}
		else_if = {
			limit = { scope:claimant = { has_weak_claim_on = title:e_britannia } }
			add = {
				value = 5
				desc = CLAIMANT_SUPPORT_UNPRESSED_CLAIM
			}
		}
		
		# Scope traits
		if = {
			limit = { has_trait = ambitious }
			subtract = {
				value = 15
				desc = CLAIMANT_SUPPORT_IS_AMBITIOUS
			}
		}
		
		# Same House/Dynasty
		if = {
			limit = { house = scope:claimant.house }
			add = {
				value = 30
				desc = CLAIMANT_SUPPORT_SAME_HOUSE
			}
		}
		else_if = {
			limit = { dynasty = scope:claimant.dynasty }
			add = {
				value = 15
				desc = CLAIMANT_SUPPORT_SAME_DYNASTY
			}
		}
		# Alliances
		if = {
			limit = { is_allied_to = scope:claimant }
			add = {
				value = 40
				desc = CLAIMANT_SUPPORT_ALLIANCE
			}
		}
		# Cultural Differences
		if = {
			limit = { this.culture != scope:claimant.culture }
			subtract = {
				value = 5
				desc = CLAIMANT_SUPPORT_DIFFERENT_CULTURE
			}
		}
		if = {
			limit = { NOT = { this.culture = { has_same_culture_heritage = scope:claimant.culture } } }
			subtract = {
				value = 10
				desc = CLAIMANT_SUPPORT_DIFFERENT_HERITAGE
			}
		}
		# Religious Differences
		if = {
			limit = {
				faith = {
					faith_hostility_level = {
						target = scope:claimant.faith
						value = faith_astray_level
					}
				}
			}
			if = {
				limit = { has_trait = zealous }
				subtract = {
					value = 15
					desc = CLAIMANT_SUPPORT_ASTRAY_FAITH_ZEALOUS
				}
			}
			else = {
				subtract = {
					value = 10
					desc = CLAIMANT_SUPPORT_ASTRAY_FAITH
				}
			}
		}
		else_if = {
			limit = {
				faith = {
					faith_hostility_level = {
						target = scope:claimant.faith
						value = faith_hostile_level
					}
				}
			}
			if = {
				limit = { has_trait = zealous }
				subtract = {
					value = 30
					desc = CLAIMANT_SUPPORT_HOSTILE_FAITH_ZEALOUS
				}
			}
			else = {
				subtract = {
					value = 20
					desc = CLAIMANT_SUPPORT_HOSTILE_FAITH
				}
			}
		}
		else_if = {
			limit = {
				faith = {
					faith_hostility_level = {
						target = scope:claimant.faith
						value = faith_evil_level
					}
				}
			}
			if = {
				limit = { has_trait = zealous }
				subtract = {
					value = 30
					desc = CLAIMANT_SUPPORT_EVIL_FAITH_ZEALOUS
				}
			}
			else = {
				subtract = {
					value = 20
					desc = CLAIMANT_SUPPORT_EVIL_FAITH
				}
			}
		}
		
		# Northern Britons don't care unless you are also a northerner
		if = {
			limit = {
				NOR = {
					scope:claimant.primary_title.title_capital_county.title_province = { 
						geographical_region = btwk_world_britannia_hen_ogledd
					}
					scope:claimant = {
						any_held_title = {
							tier >= tier_kingdom
							title_capital_county.title_province = { 
								geographical_region = btwk_world_britannia_hen_ogledd
							}
						}
					}
					primary_title.title_capital_county.title_province = { 
						geographical_region = btwk_world_britannia_south
					}
					any_held_title = {
						tier >= tier_kingdom
						title_capital_county.title_province = { 
							geographical_region = btwk_world_britannia_south
						}
					}
				}
			}
			divide = {
				value = 10
				desc = CLAIMANT_SUPPORT_DISTANT_CLAIMANT
			}
		}
		
		# Don't really care if not Brythonic or Imperial
		if = {
			limit = {
				culture = {
					NOR = {
						has_cultural_pillar = heritage_brythonic
						has_cultural_pillar = heritage_latin
					}
				}
			}
			divide = {
				value = 4
				desc = CLAIMANT_SUPPORT_NOT_BRYTHONIC_OR_IMPERIAL
			}
		}
		
		# Won't get involved if tribtuary to a non-Brythonic/Imperial
		if = {
			limit = {
				is_tributary = yes
				suzerain = {
					culture = {
						NOR = {
							has_cultural_pillar = heritage_brythonic
							has_cultural_pillar = heritage_latin
						}
					}
				}
			}
			divide = {
				value = 4
				desc = CLAIMANT_SUPPORT_SUZERAIN_NOT_BRYTHONIC_OR_IMPERIAL
			}
		} 
		
		# Don't really want to oppose if they are losing a war
		save_temporary_scope_value_as = current_score
		if = {
			limit = {
				scope:current_score < BTWK_britannia_high_king_scope_opponent_threshold
				is_at_war = yes
			}
			if = {
				limit = {
					any_character_war = {
						is_war_leader = prev
						OR = {
							AND = {
								primary_defender = prev
								attacker_war_score >= 50
							}
							AND = {
								primary_attacker = prev
								defender_war_score >= 50
							}
						}
					}
				}
				add = {
					value = 40
					desc = CLAIMANT_SUPPORT_LOSING_WAR
				}
			}
			else_if = {
				limit = {
					any_character_war = {
						is_war_leader = prev
						OR = {
							AND = {
								primary_defender = prev
								attacker_war_score >= 25
							}
							AND = {
								primary_attacker = prev
								defender_war_score >= 25
							}
						}
					}
				}
				add = {
					value = 30
					desc = CLAIMANT_SUPPORT_LOSING_WAR
				}
			}
			else_if = {
				limit = {
					any_character_war = {
						is_war_leader = prev
						OR = {
							AND = {
								primary_defender = prev
								attacker_war_score >= 10
							}
							AND = {
								primary_attacker = prev
								defender_war_score >= 10
							}
						}
					}
				}
				add = {
					value = 20
					desc = CLAIMANT_SUPPORT_LOSING_WAR
				}
			}
			else = {
				add = {
					value = 10
					desc = CLAIMANT_SUPPORT_AT_WAR
				}
			}
		}
		
		# Allies won't directly oppose you
		if = {
			limit = {
				is_allied_to = scope:claimant
			}
			min = {
				value = BTWK_britannia_high_king_scope_opponent_threshold
				desc = CLAIMANT_SUPPORT_ALLY_WILL_NOT_OPPOSE
			}
		}
	}
}

BTWK_britannia_high_king_dispute_war_military_power_ratio = {
	value = military_power
	ordered_independent_ruler = {
		limit = { has_variable = HK_britannia_opponent }
		order_by = military_power
		save_temporary_scope_as = HK_britannia_opponent_leader
	}
	# Add military power of all predicted allies
	every_independent_ruler = {
		limit = {
			this != prev
			OR = {
				is_allied_to = prev
				has_variable = HK_britannia_supporter
			}
			NOR = {
				this = scope:HK_britannia_opponent_leader
				is_allied_to = scope:HK_britannia_opponent_leader
				has_variable = HK_britannia_opponent
			}
		}
		add = military_power
	}
	# Personality adjustment - bold characters will go ahead at less favourable odds
	multiply = {
		value = 1
		add = {
			value = ai_boldness
			divide = 100
			multiply = 0.25
		}
	}
	# Divide by the military power of all predicted opponents
	divide = {
		value = 0
		every_independent_ruler = {
			limit = {
				OR = {
					this = scope:HK_britannia_opponent_leader
					is_allied_to = scope:HK_britannia_opponent_leader
					has_variable = HK_britannia_opponent
				}
			}
			add = military_power
		}		
		min = 1
	}
}

		limit = { has_legitimacy = yes }
		multiply = {
			value = BTWK_britannia_legitimacy_level_modifier
			desc = BRITANNIA_AUTHORITY_LEGITIMACY
		}
	}
	if = {
		limit = { government_has_flag = government_is_tribal }
		multiply = {
			value = 0.66
			desc = BRITANNIA_AUTHORITY_TRIBAL_GOVERNMENT
		}
	}
	else_if = {
		limit = { NOT = { government_has_flag = government_is_pre_feudal } }
		multiply = {
			value = 1.2
			desc = BRITANNIA_AUTHORITY_ADVANCED_GOVERNMENT
		}
	}
	if = {
		limit = {
			struggle:britannia_struggle ?= { has_struggle_phase_parameter = britannia_high_king_authority_minor_boost }
		}
		multiply = {
			value = BTWK_britannia_high_king_authority_minor_struggle_phase_boost_modifier
			desc = BRITANNIA_AUTHORITY_STRUGGLE_PHASE_MINOR_BOOST
		}
	}
	else_if = {
		limit = {
			struggle:britannia_struggle ?= { has_struggle_phase_parameter = britannia_high_king_authority_boost }
		}
		multiply = {
			value = BTWK_britannia_high_king_authority_struggle_phase_boost_modifier
			desc = BRITANNIA_AUTHORITY_STRUGGLE_PHASE_BOOST
		}
	}	
}

BTWK_britannia_high_king_reform_empower_high_council_support_score = {
	# Base value (generally quite reluctant)
	value = -50
	
	# Opinion
	add = "opinion(scope:high_king)"
	
	# Hooks
	if = {
		limit = { scope:high_king = { has_strong_hook = prev } }
		add = 40
	}
	else_if = {
		limit = { scope:high_king = { has_hook = prev } }
		add = 15
	}
	
	# Scope traits
	if = {
		limit = { has_trait = ambitious }
		subtract = 15
	}
	
	# Same House/Dynasty (minor boost)
	if = {
		limit = { house = scope:high_king.house }
		add = 10
	}
	else_if = {
		limit = { dynasty = scope:high_king.dynasty }
		add = 5
	}
	
	# Alliances
	if = {
		limit = { is_allied_to = scope:high_king }
		add = 20
	}
	
	# Cultural Differences
	if = {
		limit = { this.culture != scope:high_king.culture }
		subtract = 10
	}
	else_if = {
		limit = { NOT = { this.culture = { has_same_culture_heritage = scope:claimant.culture } } }
		subtract = 20
	}
	
	# Religious Differences
	subtract = {
		value = "faith.faith_hostility_level(scope:high_king.faith)"
		multiply = 10
		max = 25		# Hostile vs. Evil isn't fundamentally a big difference
		if = {
			limit = { has_trait = zealous }
			multiply = 1.5
		}
	}
}
