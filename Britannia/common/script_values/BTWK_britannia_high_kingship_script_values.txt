BTWK_britannia_high_king_scope_support_for_claimant = {
	value = 0
	add = {
		value = "opinion(scope:claimant)"
		desc = OPINION
	}
	if = {
		limit = { this.culture != scope:claimant.culture }
		subtract = {
			value = 5
			desc = DIFFERENT_CULTURE
		}
	}
	if = {
		limit = { NOT = { this.culture = { has_same_culture_heritage = scope:claimant.culture } } }
		subtract = {
			value = 10
			desc = DIFFERENT_HERITAGE
		}
	}
	if = {
		limit = {
			scope:claimant.faith = {
				faith_hostility_level = {
					target = prev.faith
					value = faith_astray_level
				}
			}
		}
		subtract = {
			value = 10
			desc = ASTRAY_FAITH
		}
	}
	else_if = {
		limit = {
			scope:claimant.faith = {
				faith_hostility_level = {
					target = prev.faith
					value = faith_hostile_level
				}
			}
		}
		subtract = {
			value = 20
			desc = HOSTILE_FAITH
		}
	}
	else_if = {
		limit = {
			scope:claimant.faith = {
				faith_hostility_level = {
					target = prev.faith
					value = faith_evil_level
				}
			}
		}
		subtract = {
			value = 30
			desc = EVIL_FAITH
		}
	}
}
