tribal_subjugation_cb_prestige_cost = {
	value = 0
	if = {
		limit = { BTWK_britannia_is_high_king_subjugation_trigger = yes }
		add = {
			value = 500
			desc = CB_BASE_COST
		}
	}
	else = {
		if = {
			limit = {
				NAND = {
					scope:attacker.faith.religious_head = scope:attacker
					scope:defender.faith = scope:attacker.faith
				}
			}
			add = {
				value = 1000
				desc = CB_BASE_COST
			}
		}
		if = {
			limit = {
				exists = scope:attacker.primary_title
				scope:attacker.primary_title = {
					tier = tier_kingdom
				}
			}
			multiply = {
				value = 1.5
				desc = CB_COST_RULES_KINGDOM
			}
		}
		else_if = {
			limit = {
				exists = scope:attacker.primary_title
				scope:attacker.primary_title = {
					tier = tier_empire
				}
			}
			multiply = {
				value = 2.5
				desc = CB_COST_RULES_EMPIRE
			}
		}
	}

	multiply = common_cb_prestige_cost_multiplier
}
