
conquests_culture_targets_ai_score_value = {
	value = 1
	# Aelle prioritizes border duchies
	if = {
		limit = { 
			scope:attacker = {
				OR = {
					has_character_flag = aelle_of_anglia
					top_liege = { has_character_flag = aelle_of_anglia }
				}
			}
		}
		every_target_title = {
			if = {
				limit = { title_province = { geographical_region = btwk_custom_aelle_target_region } }
				if = {
					limit = { tier = tier_duchy }
					add = 10
				}
				else = {
					add = 5
				}
			}
		}
	}
	# Anglo-Saxons prioritize Britannia
	if = {
		limit = {
			scope:attacker = {
				culture = { has_cultural_pillar = heritage_west_germanic }
				primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
		}
		every_target_title = {
			title_province = {
				if = {
					limit = { 
						geographical_region = btwk_world_britannia
						NOT = { geographical_region = btwk_custom_wales }
					}
					add = 3
				}
				else_if = {
					limit = { 
						geographical_region = btwk_world_britannia
					}
					add = 1
				}
			}
			# And particularly value counties that are already in e_aengland
			if = {
				limit = { empire = title:e_aengland }
				add = 2
			}
			# Also counties that are required to form Anglo-Saxon duchies
			if = {
				limit = { has_variable = anglo_saxon_duchy_requirement }
				add = 1
			}
		}
	}
}
