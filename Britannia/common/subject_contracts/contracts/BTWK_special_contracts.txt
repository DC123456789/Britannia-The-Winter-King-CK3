@ai_standard_liege_desire = 2
@ai_standard_vassal_desire = 1

tribal_obligations = {
	display_mode = checkbox
	obligation_levels = {

			position = { 0 0 }

			ai_liege_desire = @ai_standard_liege_desire
			ai_subject_desire = 0
		}
		tribal_obligations_granted = {
			is_valid = {
				scope:liege = { government_has_flag = government_has_tribal_obligations_contract } 
			}
			default = yes
			parent = tribal_obligations_none
			position = { 1 0 }

			subject_opinion = 5

			levies = {
				value = tribal_obligations_levies_baseline
				scope:liege = {
					add = {
						value = tribal_obligations_levies_per_prestige_level

					}
				}
				max = 0.5
			}
			tax = {
				value = tribal_obligations_taxes_baseline
				scope:liege = {
					add = {
						value = tribal_obligations_taxes_per_prestige_level
						multiply = prestige_level
					}
				}
				max = extortionate_feudal_tax
			}
			contribution_desc = {
				desc = "pre_feudal_government_vassal_contribution_prestige_level"
			}
			
			flag = has_tribal_obligations_contract

			ai_liege_desire = 0
			ai_subject_desire = @ai_standard_vassal_desire
		}
	}
}
