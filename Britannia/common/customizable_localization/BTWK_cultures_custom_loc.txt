GetSteleStoneRunestone = {
	type = character
	
	#Pictish
	text = {
		trigger = {
			culture = {
				has_cultural_pillar = heritage_pictish
			}
		}
		localization_key = carved_stone_stone
	}
	#Other Celtic
	text = {
		trigger = {
			culture = {
				OR = {
					has_cultural_pillar = heritage_brythonic
					has_cultural_pillar = heritage_goidelic
				}
			}
		}
		localization_key = carved_stone_stele
	}
	#Norse
	text = {
		trigger = {
			culture = {
				has_cultural_pillar = heritage_north_germanic
			}
		}
		localization_key = carved_stone_runestone
	}
	
	text = {
		localization_key = carved_stone_stone
		fallback = yes
	}
}
