	type = character

	text = {
		localization_key = count_feudal_male_brythonic_group
		trigger = {
			OR = {
				AND = {
					highest_held_title_tier < tier_county
					primary_spouse = { 
						culture = { has_same_heritage_group_as = { TARGET = culture:briton } } 
						top_liege = { 
							culture = { has_name_list = name_list_romanobritish } 
							NOT = { has_government = imperial_government }
						}
					}
				}
				AND = {
					highest_held_title_tier >= tier_county
					culture = { has_same_heritage_group_as = { TARGET = culture:briton } }
					top_liege = { 
						culture = { has_name_list = name_list_romanobritish } 
						NOT = { has_government = imperial_government }
					}
				}
			}
		}
	}
	text = {
		localization_key = generic_consort_male_latin_group
		trigger = {
			highest_held_title_tier < tier_county
		}
	}
	text = {
		localization_key = count_feudal_male_latin_group_titled
		fallback = yes

}
