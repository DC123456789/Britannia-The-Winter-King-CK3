BTWK_britannia_high_king_claim_potential_list_loc = {
	type = all
	
	text = {
		localization_key = BTWK_britannia_high_king_claim_potential_list_empty
		trigger = {
			NOT = { has_variable = value }
		}
	}
	text = {
		localization_key = BTWK_britannia_high_king_claim_potential_list_node
		trigger = {
			has_variable = value
			has_variable = next
		}
	}
	text = {
		localization_key = BTWK_britannia_high_king_claim_potential_list_terminal_node
		trigger = {
			has_variable = value
			NOT = { has_variable = next }
		}
	}
}

BTWK_HK_reform_centralize_the_church_name = {
	type = character
	
	text = {
		localization_key = BTWK_HK_reform_centralize_the_church_christian
		trigger = {
			title:e_britannia.holder ?= { has_religion = religion:christianity_religion }
		}
	}
	text = {
		localization_key = BTWK_HK_reform_centralize_the_church_nonchristian
		trigger = {
			title:e_britannia.holder ?= { NOT = { has_religion = religion:christianity_religion } }
		}
	}
	text = {
		localization_key = BTWK_HK_reform_centralize_the_church_christian
		trigger = {
			NOT = { exists = title:e_britannia.holder }
			has_religion = religion:christianity_religion
		}
	}
	text = {
		localization_key = BTWK_HK_reform_centralize_the_church_nonchristian
		trigger = {
			NOT = { exists = title:e_britannia.holder }
			NOT = { has_religion = religion:christianity_religion }
		}
	}
}
