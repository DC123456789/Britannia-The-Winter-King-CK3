BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story_trigger = {
	custom_description = {
		text = BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story
		subject = $TARGET$
		# Variable exists and story is active (has not been ended)
		$TARGET$.var:rise_of_anglo_saxon_kingdom_story ?= { always = yes }		
	}
}

BTWK_rise_of_anglo_saxon_kingdom_time_trigger = {
	trigger_if = {
		limit = { is_ai = no }
		# Human players get a slight head start so it's easier for them to start the story themselves
		custom_tooltip = {
			text = BTWK_rise_of_anglo_saxon_kingdom_decision_year_tt
			current_year >= 530
		}
	}
	trigger_else = {
		current_year >= 532
	}
	custom_tooltip = {
		text = BTWK_britannia_struggle_not_detente_phase
		struggle:britannia_struggle ?= { has_struggle_phase_parameter = allow_rise_of_anglo_saxon_kingdoms }
	}
}

BTWK_rise_of_anglo_saxon_kingdom_get_target_kingdom_trigger = {
	trigger_if = {
		limit = {
			primary_title.title_capital_county.var:anglo_saxon_duchy.kingdom = title:k_bernicia
		}
		title:k_bernicia = { save_temporary_scope_as = target_kingdom }
	}
	trigger_else_if = {
		limit = {
			primary_title.title_capital_county.var:anglo_saxon_duchy.kingdom = title:k_deira
			primary_title.title_capital_county.var:anglo_saxon_duchy != title:d_lindsey
		}
		title:k_deira = { save_temporary_scope_as = target_kingdom }
	}
	trigger_else_if = {
		limit = {
			primary_title.title_capital_county.var:anglo_saxon_duchy = title:d_lindsey
		}
		title:k_lindsey = { save_temporary_scope_as = target_kingdom }
	}
	trigger_else_if = {
		limit = {
			primary_title.title_capital_county.var:anglo_saxon_duchy.kingdom = title:k_mercia
		}
		title:k_mercia = { save_temporary_scope_as = target_kingdom }
	}
	trigger_else_if = {
		limit = {
			primary_title.title_capital_county.var:anglo_saxon_duchy.kingdom = title:k_wessex
		}
		title:k_wessex = { save_temporary_scope_as = target_kingdom }
	}
	trigger_else = {
		always = no
	}
}

BTWK_rise_of_anglo_saxon_kingdom_is_kingdom_visible_trigger = {
	trigger_if = {
		limit = { $KINGDOM$ = title:k_bernicia }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_bernicia } }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_northumbria } }
	}
	trigger_else_if = {
		limit = { $KINGDOM$ = title:k_deira }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_deira } }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_northumbria } }
	}
	trigger_else_if = {
		limit = { $KINGDOM$ = title:k_lindsey }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_lindsey } }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:d_lindsey } }
	}
	trigger_else_if = {
		limit = { $KINGDOM$ = title:k_mercia }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_mercia } }
	}
	trigger_else_if = {
		limit = { $KINGDOM$ = title:k_wessex }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_wessex } }
	}
	trigger_else = {
		always = no
	}
}

		NOT = { BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story_trigger = { TARGET = title:k_lindsey } }
	}
	trigger_else_if = {
		limit = { $KINGDOM$ = title:k_mercia }
		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_mercia } }
		NOT = { BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story_trigger = { TARGET = title:k_mercia } }

		NOT = { BTWK_title_has_been_created_trigger = { TARGET = title:k_wessex } }
		NOT = { BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story_trigger = { TARGET = title:k_wessex } }

}

BTWK_rise_of_anglo_saxon_kingdom_is_character_eligible_excluding_temp_factors_trigger = {
	BTWK_rise_of_anglo_saxon_kingdom_time_trigger = yes
	is_unromanized_germanic_culture_trigger = yes
	BTWK_rise_of_anglo_saxon_kingdom_get_target_kingdom_trigger = yes
	OR = {
		is_independent_ruler = yes
		custom_tooltip = {
			text = BTWK_rise_of_anglo_saxon_kingdom_decision_liege_culture_tt
			NOT = { top_liege = { is_unromanized_germanic_culture_trigger = yes } }
		}
		AND = {
			primary_title.title_capital_county.var:anglo_saxon_duchy.kingdom = {
				save_temporary_scope_as = de_jure_kingdom
			}
			custom_tooltip = {
				text = BTWK_rise_of_anglo_saxon_kingdom_decision_liege_other_kingdom_tt
				top_liege = {
					highest_held_title_tier >= tier_kingdom
					NOT = { has_title = scope:de_jure_kingdom }
				}
			}
		}
	}
	highest_held_title_tier <= tier_duchy
	BTWK_rise_of_anglo_saxon_kingdom_is_kingdom_eligible_trigger = { KINGDOM = scope:target_kingdom }
}

BTWK_rise_of_anglo_saxon_kingdom_is_character_eligible_trigger = {
	BTWK_rise_of_anglo_saxon_kingdom_is_character_eligible_excluding_temp_factors_trigger = yes
	is_at_war = no
	is_imprisoned = no
}
