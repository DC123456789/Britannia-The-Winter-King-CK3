#Triggers to control Coat of Arms weighting

coa_roman_romanized_trigger = {
	scope:culture = {
		has_coa_gfx = latin_group_coa_gfx #Inc. Alan and Romanized Germanic
	}
}

coa_pictish_trigger = {
	OR = {
		scope:faith = { has_icon = "pictish_pagan_reformed" }
		scope:faith = { has_icon = "pictish_pagan" }
		scope:culture = { has_name_list = name_list_pictish }
	}
}

coa_taifal_trigger = {
	scope:culture = {
		OR = { 
			has_name_list = name_list_taifal
			culture_is_culture_or_descendant_including_romance_trigger = { CULTURE = culture:taifal }
		}
	}
}

coa_gothic_trigger = {
	scope:culture = {
		OR = { 
			has_name_list = name_list_visigoth
			culture_is_culture_or_descendant_including_romance_trigger = { CULTURE = culture:visigoth }
			has_name_list = name_list_ostrogoth
			culture_is_culture_or_descendant_including_romance_trigger = { CULTURE = culture:ostrogoth }
		}
	}
}

coa_high_german_trigger = {
	scope:culture = {
		has_coa_gfx = high_german_coa_gfx
	}
}

coa_switch_to_christian_trigger = {
	holder = {
		faith.religion = religion:christianity_religion
	}
}

coa_switch_to_non_christian_trigger = {
	holder = {
		NOT = { faith.religion = religion:christianity_religion }
	}
}
