is_anglo_saxon_heptarchy_kingdom_trigger = {
	is_target_in_global_variable_list = {
		name = anglo_saxon_heptarchy_kingdom_list
		target = this
	}
}

is_anglo_saxon_kingdom_trigger = {
	is_target_in_global_variable_list = {
		name = anglo_saxon_kingdom_list
		target = this
	}
}

BTWK_title_has_been_created_trigger = {
	custom_description = {
		text = BTWK_title_has_been_created
		subject = $TARGET$
		$TARGET$ = { has_variable = title_has_been_created }
	}
}

BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story_trigger = {
	custom_description = {
		text = BTWK_title_has_ongoing_rise_of_anglo_saxon_kingdom_story
		subject = $TARGET$
		# Variable exists and story is active (has not been ended)
		$TARGET$.var:rise_of_anglo_saxon_kingdom_story ?= { always = yes }		
	}
}
