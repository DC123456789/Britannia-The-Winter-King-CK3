BTWK_is_compatible_culture_with_cultural_redoubt_trigger = {
	trigger_if = {
		limit = {
			$COUNTY$.title_province = { geographical_region = btwk_world_britannia }
		}
		# Is not Celtic or Roman/Romanized
		NOR = {
			culture = { has_same_heritage_group_as = { TARGET = culture:briton } }
			is_imperial_culture_trigger = yes
			is_romanized_barbarian_culture_trigger = yes
			culture = { has_cultural_pillar = heritage_romance }
		}
	}
}
