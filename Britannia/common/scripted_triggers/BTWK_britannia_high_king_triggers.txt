BTWK_britannia_high_kingship_can_claim_trigger = {
	is_independent_ruler = yes
	custom_description = {
		text = claim_britannia_high_kingship_culture_requirement_w_britannic
		BTWK_britannia_high_kingship_is_allowed_culture_trigger = yes
	}
	custom_description = {
		text = claim_britannia_high_kingship_kingdom_requirement
		OR = {
			any_held_title = {
				tier = tier_kingdom
				target_is_de_jure_liege_or_above = title:e_britannia
				title_capital_county.title_province = { geographical_region = btwk_world_britannia_south }
			}
			has_title = title:e_hen_ogledd
		}
	}
	title:e_britannia = { is_title_created = no }
	prestige_level >= high_prestige_level
	age >= 25
	realm_size >= 10
	is_tributary = no
	custom_description = {
		text = claim_britannia_high_kingship_military_requirement
		current_military_strength >= 750
	}
	diplomacy >= 10
	martial >= 12
	stewardship >= 7
	custom_description = {
		text = claim_britannia_high_kingship_skill_requirement
		OR = {
			any_held_title = {
				tier >= tier_kingdom
				title_held_years >= 15 
			}
			prestige_level >= 4
			diplomacy >= 25
			martial >= 25
			prowess >= 30
		}
	}
	is_incapable = no
	NOT = { has_diarchy_type = regency }
	is_at_war = no
	custom_description = {
		text = claim_britannia_high_kingship_no_current_claimant
		NOT = { title:e_britannia = { has_variable = britannia_high_kingship_claimant } }
	}
	trigger_if = {
		limit = { has_character_modifier = BTWK_recently_claimed_high_kingship_modifier }
		NOT = { has_character_modifier = BTWK_recently_claimed_high_kingship_modifier }
	}
	trigger_if = {
		limit = { has_character_modifier = BTWK_was_recently_high_king_modifier }
		NOT = { has_character_modifier = BTWK_was_recently_high_king_modifier }
	}
}

BTWK_britannia_high_kingship_is_allowed_culture_trigger = {
	custom_description = {
		text = BTWK_britannia_high_kingship_is_allowed_culture
		culture = {
			OR = {
				has_cultural_pillar = heritage_brythonic
				has_cultural_pillar = heritage_latin
				AND = {
					has_cultural_pillar = heritage_romance
					NOT = { has_variable = ancestral_non_roman_culture }
					var:ancestral_starting_culture ?= culture:romanobritish
				}
			}
		}
	}
}
