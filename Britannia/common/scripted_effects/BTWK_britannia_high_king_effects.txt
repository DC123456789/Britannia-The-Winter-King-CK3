# Sets up all the necessary variables for viewing the High King window and its associated tooltips
setup_britannia_high_king_window_lists_effect = {
	save_scope_as = claimant

	# Clear old lists
	clear_global_variable_list = britannia_subject_kings_list
	clear_global_variable_list = britannia_independent_kings_list
	clear_variable_list = britannia_potential_supporter_kings_list
	clear_variable_list = britannia_potential_opponent_kings_list
	clear_variable_list = britannia_potential_neutral_kings_list
	clear_global_variable_list = britannia_supporter_kings_list
	clear_global_variable_list = britannia_opponent_kings_list
	clear_global_variable_list = britannia_neutral_kings_list
	
	# These stories are also cleared on the death of the owner to avoid bloat
	every_in_list = {
		variable = britannia_potential_supporter_kings_linked_list
		end_story = yes
	}
	clear_variable_list = britannia_potential_supporter_kings_linked_list
	every_in_list = {
		variable = britannia_potential_opponent_kings_linked_list
		end_story = yes
	}
	clear_variable_list = britannia_potential_opponent_kings_linked_list
	every_in_list = {
		variable = britannia_potential_neutral_kings_linked_list
		end_story = yes
	}
	clear_variable_list = britannia_potential_neutral_kings_linked_list

	every_ruler = {
		limit = {
			OR = {
				BTWK_britannia_high_kingship_is_elector_trigger = yes
				BTWK_britannia_high_kingship_is_subject_king_vassal_or_tributary_trigger = yes
			}
		}
		add_to_temporary_list = britannia_all_kings_list
	}
	
	ordered_in_list = {
		list = britannia_all_kings_list
		max = list_size:britannia_all_kings_list
		order_by = sub_realm_size

		# Sort independent vs. subject kings
		if = {
			limit = {
				OR = {
					top_liege = { has_title = title:e_britannia }
					AND = {
						is_tributary = yes
						BTWK_britannia_high_kingship_is_subject_king_tributary_trigger = yes
						suzerain = { has_title = title:e_britannia }
					}
				}
			}
			if = {
				limit = { NOT = { has_title = title:e_britannia } }		# High king himself shouldn't be here
				add_to_global_variable_list = {
					name = britannia_subject_kings_list
					target = THIS
				}
			}
		}
		else = {
			add_to_global_variable_list = {
				name = britannia_independent_kings_list
				target = THIS
			}
		}
		# Calculate potential supporter vs. opponent vs. neutral kings
		if = {
			limit = { this != scope:claimant }
			if = {
				limit = { 
					BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_supporter_threshold
				}
				scope:claimant = {
					add_to_variable_list = {
						name = britannia_potential_supporter_kings_list
						target = PREV
					}
				}
			}
			else_if = {
				limit = { 
					BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_opponent_threshold
				}
				scope:claimant = {
					add_to_variable_list = {
						name = britannia_potential_neutral_kings_list
						target = PREV
					}
				}
			}
			else = {
				scope:claimant = {
					add_to_variable_list = {
						name = britannia_potential_opponent_kings_list
						target = PREV
					}
				}
			}
		}
		
		# Sort declared supporter vs. opponent vs. neutral kings
		if = {
			limit = {
				has_variable = HK_britannia_supporter
			}
			add_to_global_variable_list = {
				name = britannia_supporter_kings_list
				target = THIS
			}
		}
		else_if = {
			limit = {
				has_variable = HK_britannia_opponent
			}
			add_to_global_variable_list = {
				name = britannia_opponent_kings_list
				target = THIS
			}
		}
		else_if = {
			limit = {
				OR = {
					NOT = { exists = title:e_britannia.var:britannia_high_kingship_claimant }
					title:e_britannia.var:britannia_high_kingship_claimant != this
				}
			}
			add_to_global_variable_list = {
				name = britannia_neutral_kings_list
				target = THIS
			}
		}
	}
	
	if = {
		limit = {
			OR = {
				NOT = { has_global_variable = last_high_king_window_view_num }
				global_var:BTWK_last_high_king_window_view_num = 10
			}
		}
		set_global_variable = { name = BTWK_last_high_king_window_view_num value = 1 }
		set_global_variable = { name = BTWK_high_king_claimant_1 value = scope:claimant }
		scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 1 } }
	}
	else = {
		change_global_variable = { name = BTWK_last_high_king_window_view_num add = 1 }
		if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 2 }
			set_global_variable = { name = BTWK_high_king_claimant_2 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 2 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 3 }
			set_global_variable = { name = BTWK_high_king_claimant_3 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 3 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 4 }
			set_global_variable = { name = BTWK_high_king_claimant_4 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 4 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 5 }
			set_global_variable = { name = BTWK_high_king_claimant_5 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 5 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 6 }
			set_global_variable = { name = BTWK_high_king_claimant_6 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 6 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 7 }
			set_global_variable = { name = BTWK_high_king_claimant_7 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 7 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 8 }
			set_global_variable = { name = BTWK_high_king_claimant_8 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 8 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 9 }
			set_global_variable = { name = BTWK_high_king_claimant_9 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 9 } }
		}
		else_if = {
			limit = { global_var:BTWK_last_high_king_window_view_num = 10 }
			set_global_variable = { name = BTWK_high_king_claimant_10 value = scope:claimant }
			scope:claimant = { set_variable = { name = BTWK_high_king_window_view_num value = 10 } }
		}
	}
	if = {
		limit = { 
			exists = title:e_britannia.var:britannia_high_kingship_claimant
			title:e_britannia.var:britannia_high_kingship_claimant != scope:claimant
		}
		title:e_britannia.var:britannia_high_kingship_claimant = {
			save_scope_as = claimant
			clear_variable_list = britannia_potential_supporter_kings_list
			clear_variable_list = britannia_potential_opponent_kings_list
			clear_variable_list = britannia_potential_neutral_kings_list
			ordered_in_list = {
				list = britannia_all_kings_list
				max = list_size:britannia_all_kings_list
				order_by = sub_realm_size
				# If there is an actual claimant, calculate potential stances for them too
				if = {
					limit = {
						title:e_britannia.var:britannia_high_kingship_claimant != this
					}
					if = {
						limit = { 
							BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_supporter_threshold
						}
						title:e_britannia.var:britannia_high_kingship_claimant = {
							add_to_variable_list = {
								name = britannia_potential_supporter_kings_list
								target = PREV
							}
						}
					}
					else_if = {
						limit = { 
							BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_opponent_threshold
						}
						title:e_britannia.var:britannia_high_kingship_claimant = {
							add_to_variable_list = {
								name = britannia_potential_neutral_kings_list
								target = PREV
							}
						}
					}
					else = {
						title:e_britannia.var:britannia_high_kingship_claimant = {
							add_to_variable_list = {
								name = britannia_potential_opponent_kings_list
								target = PREV
							}
						}
					}
				}
			}
		}
	}
}

BTWK_britannia_initialize_high_kingship_effect = {
	dynasty = { 
		if = {
			limit = { NOT = { has_variable = britannia_high_kingship_authority } }
			set_variable = { name = britannia_high_kingship_authority value = 0 }
		}
	}
	if = {
		limit = { NOT = { owns_story_of_type = BTWK_britannia_high_kingship_story } }
		if = {
			# This check returns false if the high_kingship_story variable exists but points to
			# an empty/ended story, which shouldn't ever happen but is good to check against anyways
			limit = { title:e_britannia.var:high_kingship_story ?= { always = yes } }
			title:e_britannia.var:high_kingship_story = { make_story_owner = prev }
		}
		else = {
			create_story = BTWK_britannia_high_kingship_story
		}
	}
}

BTWK_britannia_high_kingship_check_king_tier_vassals_effect = {
	# Check if is a potential subject kingdom
	if = {
		limit = {
			highest_held_title_tier >= tier_kingdom
			is_independent_ruler = no
			liege = {
				has_title = title:e_britannia
			}
			BTWK_britannia_high_kingship_active_trigger = yes
		}
		# Check if should be converted into a subject kingdom tributary
		if = {
			limit = {
				BTWK_britannia_subject_king_tributaries_active_trigger = yes
				NOT = {
					liege = {
						any_held_title = {
							this != title:e_britannia
							tier >= tier_empire
						}
					}
				}
			}
			liege = { save_scope_as = former_liege }
			create_title_and_vassal_change = {
				type = independency
				save_scope_as = change
				add_claim_on_loss = yes
			}
			becomes_independent = {
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change
			BTWK_britannia_high_kingship_make_subject_king_tributary_effect = { SUZERAIN = scope:former_liege }
		}
		# Check if should be converted into a subject kingdom vassal
		else_if = {
			limit = {
				BTWK_britannia_subject_king_vassals_forced_trigger = yes
				NOT = { vassal_contract_has_flag = has_subject_kingdom_vassal_contract }
				OR = {
					NOT = {
						liege = {
							any_held_title = {
								this != title:e_britannia
								tier >= tier_empire
							}
						}
					}
					liege = { save_temporary_scope_as = liege_scope }
					primary_title = {
						title_capital_county.title_province = {
							geographical_region = btwk_world_britannia
						}
						NOT = {
							empire.holder ?= {
								this = scope:liege_scope
							}
						}
					}
				}
			}
			BTWK_britannia_high_kingship_make_subject_king_vassal_effect = yes
		}
	}	
}

BTWK_britannia_high_kingship_make_subject_king_tributary_effect = {
	if = {
		limit = { BTWK_britannia_high_kingship_can_inherit_trigger = yes }
		start_tributary = {
			contract_group = tributary_britannia_subject_king_permanent
			suzerain = $SUZERAIN$
		}
	}
	else = {
		start_tributary = {
			contract_group = tributary_britannia_subject_king
			suzerain = $SUZERAIN$
		}
	}
}

	}

}
