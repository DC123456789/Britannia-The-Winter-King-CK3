apply_tribal_fame_effect = {
	every_ruler = {
		remove_tribal_fame_effect = yes
		if = {
			limit = {
				government_has_flag = government_has_tribal_fame
				tribal_fame_value >= 0
			}
			save_scope_as = current_character
			add_character_modifier = {
				modifier = tribal_fame_positive
				desc = tribal_fame_positive_desc
			}
		}
		if = {
			limit = {
				government_has_flag = government_has_tribal_fame
				tribal_fame_value < 0
			}
			save_scope_as = current_character
			add_character_modifier = {
				modifier = tribal_fame_negative
				desc = tribal_fame_negative_desc
			}
		}
	}
}

tribal_fame_government_change_effect = {
	root = {
		if = {
			limit = {
				government_has_flag = government_has_tribal_fame
			}
			save_scope_as = current_character
			add_character_modifier = {
				modifier = tribal_fame_positive
				desc = tribal_fame_positive_desc
			}
		}
		if = {
			limit = {
				NOT = { government_has_flag = government_has_tribal_fame }
			}
			remove_tribal_fame_effect = yes
		}
	}
}

remove_tribal_fame_effect = {
	if = {
		limit = {
			has_character_modifier = tribal_fame_positive
		}
		remove_character_modifier = tribal_fame_positive
	}
	if = {
		limit = {
			has_character_modifier = tribal_fame_negative
		}
		remove_character_modifier = tribal_fame_negative
	}
}
