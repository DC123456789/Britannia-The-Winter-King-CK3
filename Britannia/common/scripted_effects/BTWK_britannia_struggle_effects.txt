BTWK_end_britannia_struggle_effect = {
	if = {
		limit = { 
			exists = title:e_britannia.holder
			BTWK_britannia_high_kingship_active_trigger = yes
		}
		title:e_britannia.holder = { BTWK_britannia_high_kingship_destroy_effect = yes }
	}
	if = {
		limit = { 
			exists = title:e_aengland.holder
			BTWK_aengland_bretwalda_active_trigger = yes
		}
		title:e_aengland.holder = { BTWK_aengland_bretwalda_destroy_effect = yes }
	}
}

# Requires that the target_character and current_story scopes be set
BTWK_rise_of_anglo_saxon_kingdom_spawn_ai_expansion_war_army_effect = {
	if = {
		limit = {
			scope:target_character = {
				current_military_strength > prev.current_military_strength
			}
		}
		spawn_army = {
			name = RISE_OF_ANGLO_SAXON_KINGDOM_WARRIORS
			levies = 300
			men_at_arms = {
				type = pikemen_unit
				men = 100
			}
			men_at_arms = {
				type = landsknecht
				men = 75
			}
			location = this.capital_province
			save_scope_as = BTWK_rise_of_new_anglo_saxon_kingdom_story_army
		}
		scope:BTWK_rise_of_new_anglo_saxon_kingdom_story_army = {
			set_variable = {
				name = BTWK_rise_of_new_anglo_saxon_kingdom_story_army
				value = scope:current_story
			}
		}
	}
	else_if = {
		limit = {
			scope:target_character = {
				current_military_strength > prev.eighty_percent_of_current_military_strength
			}
		}
		spawn_army = {
			name = RISE_OF_ANGLO_SAXON_KINGDOM_WARRIORS
			levies = 150
			men_at_arms = {
				type = pikemen_unit
				men = 50
			}
			men_at_arms = {
				type = landsknecht
				men = 25
			}
			location = this.capital_province
			save_scope_as = BTWK_rise_of_new_anglo_saxon_kingdom_story_army
		}
		scope:BTWK_rise_of_new_anglo_saxon_kingdom_story_army = {
			set_variable = {
				name = BTWK_rise_of_new_anglo_saxon_kingdom_story_army
				value = scope:current_story
			}
		}
	}
	# If we're getting too weak, spawn more event troops!
	BTWK_rise_of_anglo_saxon_kingdom_spawn_ai_war_army_if_weak_effect = yes
}
