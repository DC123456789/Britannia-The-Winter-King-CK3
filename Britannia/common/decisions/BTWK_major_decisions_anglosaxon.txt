			FRACTION_REQUIRED = 0.75
		}
		custom_tooltip = {
			text = BTWK_HK_aengland_declare_supremacy_decision_hegemony_score_tt
			this.var:aengland_bretwalda_hegemony_score >= 500
		}
	}

	effect = {
		custom_tooltip = BTWK_HK_aengland_declare_supremacy_decision_effect_tt.1
		custom_tooltip = BTWK_HK_aengland_declare_supremacy_decision_effect_tt.2
		save_scope_as = bretwalda

		every_player = {
			limit = { this != prev }
			trigger_event = { id = BTWK_aengland_bretwalda.0002 }
		}
	}

	}

		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
	
	widget = {
		gui = "decision_view_widget_view_aengland_bretwalda"

embrace_anglo_saxon_culture_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_legitimacy.dds"
	}
	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 0
		empire = 36
		hegemony = 36
	}

	desc = embrace_anglo_saxon_culture_decision_desc

	is_shown = {
		has_title = title:e_aengland
		BTWK_aengland_bretwalda_has_declared_supremacy_trigger = yes
		NOT = { title:e_aengland = { has_variable = anglo_saxon_unified_culture } }
		culture = { has_cultural_pillar = heritage_west_germanic }
	}

	is_valid = {
		is_independent_ruler = yes
		government_is_not_tribal_trigger = yes
		custom_tooltip = {
			text = follows_reformed_faith_tt
			faith = { NOT = { has_doctrine_parameter = unreformed } }
		}
		prestige_level >= medium_prestige_level
		title:e_aengland = { title_held_years >= 30 }
		# TODO Easier if many Anglo-Saxon provinces are under foreign (i.e. Danish) rule?
		control_region_fraction_trigger = {
			FRACTION_REQUIRED = 0.75
			REGION = geographical_region:btwk_custom_aengland
		}
		OR = {
			custom_tooltip = {
				text = aengland_has_been_united_tt
				# TODO check variable once mechanic is implemented
				title:e_aengland = { has_variable = aengland_united }
			}
			custom_tooltip = {
				text = embrace_anglo_saxon_culture_decision_hegemony_score_requirement_tt
				always = no
			}
		}
	}

	effect = {
		add_prestige = 300
		if = {
			limit = { capital_county.culture = { has_cultural_pillar = heritage_west_germanic } }
			custom_tooltip = { text = embrace_anglo_saxon_culture_decision_tt }
		}
		else = {
			custom_tooltip = { text = embrace_anglo_saxon_culture_decision_no_capital_tt }
		}
		custom_tooltip = { text = embrace_anglo_saxon_culture_decision_adopt_anglo_saxon_tt }
		trigger_event = { id = culture_shifts.0020 }
	}
	
	ai_potential = {
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
}

BTWK_rise_of_anglo_saxon_kingdom_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/ep3_decision_varangian_guard.dds"
	}
	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 0
		empire = 0
		hegemony = 0
	}

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					primary_title = title:d_laamp_scyldings
				}
				desc = BTWK_rise_of_anglo_saxon_kingdom_decision_d_laamp_scyldings_desc
			}
			desc = BTWK_rise_of_anglo_saxon_kingdom_decision_desc
		}
	}

	is_shown = {
		BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
		BTWK_rise_of_anglo_saxon_kingdom_get_target_kingdom_trigger = yes
		BTWK_rise_of_anglo_saxon_kingdom_is_kingdom_visible_trigger = { KINGDOM = scope:target_kingdom }
	}

	is_valid = {
		BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_trigger = yes
		is_adult = yes
		prestige_level >= 2
		trigger_if = {
			limit = { is_tributary = yes }
			NOT = { has_truce = suzerain }
		}
		trigger_if = {
			limit = { is_independent_ruler = no }
			NOT = { has_truce = top_liege }
		}
	}

	effect = {
		custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.1
		show_as_tooltip = {
			spawn_army = {
				name = RISE_OF_ANGLO_SAXON_KINGDOM_WARRIORS
				levies = 300
				men_at_arms = {
					type = pikemen_unit
					men = 150
				}
				men_at_arms = {
					type = landsknecht
					men = 150
				}
				men_at_arms = {
					type = onager
					men = 10
				}
				location = this.capital_province
			}
		}
		if = {
			limit = { 
				is_independent_ruler = no
				top_liege = { realm_size <= 12 }
			}
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.2.a
		}
		else_if = {
			limit = { is_independent_ruler = no }
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.2.b
		}
		else_if = {
			limit = { is_tributary = yes }
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.2.c
		}
		else_if = {
			limit = { 
				is_landless_adventurer = yes
				primary_title = title:d_laamp_scyldings
			}
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.2.d
		}
		if = {
			limit = { 
				primary_title.title_capital_county.var:anglo_saxon_duchy = title:d_lindsey
			}
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.3.lindsey
		}
		else = {
			custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.3
		}
		custom_tooltip = BTWK_rise_of_anglo_saxon_kingdom_decision_effect_tt.4
		trigger_event = BTWK_anglo_saxon_invasions.0110
	}
	
	cost = {
		prestige = 250
	}
	
	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0		# Handled by event
	}
}
