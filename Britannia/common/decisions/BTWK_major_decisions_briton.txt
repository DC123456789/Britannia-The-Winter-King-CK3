		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
		modifier = {
			factor = 0
			BTWK_britannia_high_king_dispute_war_predicted_military_power_ratio < 1.1
		}		
	}
	
	widget = {
		gui = "decision_view_widget_view_britannia_high_kingship"
