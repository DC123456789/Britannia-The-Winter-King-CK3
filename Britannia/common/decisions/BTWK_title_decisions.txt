create_anglo_saxon_de_jure_duchies_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	major = yes

	ai_check_interval = 1

	desc = create_anglo_saxon_de_jure_duchies_decision_desc

	is_shown = {
		can_create_germanic_title_trigger = yes
		any_in_global_list = {
			variable = anglo_saxon_duchy_list
			NOT = { has_variable = title_created_de_jure }
		}
	}

	is_valid_showing_failures_only = {
		trigger_if = {
			limit = {
				is_ai = no
				NOT = { has_variable = selected_anglo_saxon_duchy }
			}
			custom_tooltip = {
				text = create_anglo_saxon_de_jure_duchies_decision_selected_duchy_tt
				has_variable = selected_anglo_saxon_duchy
			}
		}
		trigger_if = {
			limit = {
				is_ai = no
				has_variable = selected_anglo_saxon_duchy
			}
			completely_controls = this.var:selected_anglo_saxon_duchy.title_capital_county
		}
	}

	effect = {
		if = {
			limit = {
				is_ai = no
				has_variable = selected_anglo_saxon_duchy
			}
			show_as_tooltip = {
				create_title_and_vassal_change = {
					type = created
					save_scope_as = title_change
					add_claim_on_loss = no
				}
				this.var:selected_anglo_saxon_duchy = {
					change_title_holder = {
						holder = root
						change = scope:title_change
					}
				}
				resolve_title_and_vassal_change = scope:title_change
			}
			this.var:selected_anglo_saxon_duchy = {
				every_in_list = {
					variable = dynamic_de_jure_territory
					show_as_tooltip = {
						set_de_jure_liege_title = prev
					}
					# custom_tooltip = create_anglo_saxon_de_jure_duchies_decision.duchy_province_effect.tt
				}
				save_scope_as = new_duchy_title
				set_variable = title_created_de_jure
			}
			hidden_effect = {
				save_scope_as = conqueror
				trigger_event = { id = titular_titles.0001 }
				every_player = {
					limit = { NOT = { this = root } }
					trigger_event = { id = titular_titles.0001 }
				}
				remove_variable = selected_anglo_saxon_duchy
			}
		}
	}
	
	ai_potential = {
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
	
	widget = {
		gui = "decision_view_widget_create_anglo_saxon_de_jure_duchies"
		controller = decision_option_list_controller

		### Increase Unity
		item = {
			value = steer_unity_towards_harmonious
			is_valid = {
				custom_tooltip = {
					text = steer_unity_towards_harmonious_tt
					NOT = { house = { has_house_unity_stage = harmonious } }
				}
			}
			current_description = steer_unity_towards_harmonious_desc
			localization = steer_unity_towards_harmonious_name
			icon = "gfx/interface/icons/currencies/house_unity/harmonious.dds"
			ai_chance = {
				value = 100
				if = {
					limit = { ai_wants_low_unity = yes }
					multiply = 0
				}
			}
		}
	}
}
