create_roman_de_jure_duchies_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	decision_group_type = major

	ai_check_interval = 1

	desc = create_roman_de_jure_duchies_decision_desc

	is_shown = {
		can_create_roman_title_show_trigger = yes
		any_in_global_list = {
			variable = roman_duchy_list
			NOT = { has_variable = roman_title_created_de_jure }
		}
	}

	is_valid_showing_failures_only = {
		can_create_roman_title_trigger = yes
		trigger_if = {
			limit = { is_independent_ruler = no }
			liege = { highest_held_title_tier > tier_duchy }
		}
		custom_tooltip = {
			text = create_roman_de_jure_duchies_decision_any_formable_tt
			any_in_global_list = {
				variable = roman_duchy_list
				NOT = { has_variable = roman_title_created_de_jure }
				ROOT = { can_create_formable_title = { TITLE = PREV } }
			}
		}
	}

	effect = {
		# This part is purely for display purposes
		if = {
			limit = {
				is_ai = no
				has_variable = selected_title
			}
			this.var:selected_title = {
				save_scope_as = roman_title_created_de_jure_pretend
				show_as_tooltip = {
					check_de_jure_liege_duchy_effect = yes
				}
				every_in_list = {
					variable = roman_de_jure_territory
					show_as_tooltip = {
						check_de_jure_liege_county_effect = yes
					}
				}
			}
			show_as_tooltip = {
				if = {
					limit = {
						exists = this.var:selected_title.holder
						NOR = {
							this.var:selected_title.holder = ROOT
							this.var:selected_title.holder = { is_vassal_or_below_of = ROOT }
						}
					}			
					create_title_and_vassal_change = {
						type = usurped
						save_scope_as = title_change
						add_claim_on_loss = yes
					}
					this.var:selected_title = {
						change_title_holder = {
							holder = root
							change = scope:title_change
						}
					}
					resolve_title_and_vassal_change = scope:title_change
				}
				else_if = {
					limit = { NOT = { exists = this.var:selected_title.holder } }
					create_title_and_vassal_change = {
						type = created
						save_scope_as = title_change
						add_claim_on_loss = no
					}
					this.var:selected_title = {
						change_title_holder = {
							holder = root
							change = scope:title_change
						}
					}
					resolve_title_and_vassal_change = scope:title_change
				}
			}
		}
		hidden_effect = {
			every_in_global_list = {
				variable = roman_duchy_list
				limit = { 
					ROOT = { can_create_formable_title = { TITLE = PREV } }
					NOT = { has_variable = roman_title_created_de_jure }
				}
				ROOT = { create_roman_duchy_effect = { TITLE = PREV } }
			}
			remove_variable = selected_title
		}
	}
	
	ai_potential = {
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
	
	widget = {
		gui = "decision_view_widget_create_roman_de_jure_duchies"
		controller = decision_option_list_controller

		### This placeholder has to be here or the game will crash
		item = {
			value = none
			is_shown = { always = yes }
			is_valid = { always = yes }
			localization = placeholder_name
		}
	}
}

destroy_roman_de_jure_duchies_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	decision_group_type = major

	ai_check_interval = 1

	desc = destroy_roman_de_jure_duchies_decision_desc

	is_shown = {
		can_create_roman_title_show_trigger = no
		can_create_germanic_title_culture_trigger = no
		any_in_global_list = {
			variable = roman_duchy_list
			has_variable = roman_title_created_de_jure
		}
	}

	is_valid_showing_failures_only = {
		NOT = { top_liege_can_create_roman_title_trigger = yes }
		NOT = { top_liege_can_create_germanic_title_culture_trigger = yes }
		custom_tooltip = {
			text = destroy_roman_de_jure_duchies_decision_any_destroyable_tt
			any_in_global_list = {
				variable = roman_duchy_list
				has_variable = roman_title_created_de_jure
				ROOT = {
					can_destroy_formable_title = { 
						TITLE = PREV
						COUNTY_TRIGGER = "NOT = { roman_controlled_trigger = yes }" 
					}
				}
			}
		}
	}

	effect = {
		# This part is purely for display purposes
		if = {
			limit = {
				is_ai = no
				has_variable = selected_title
			}
			this.var:selected_title = {
				save_scope_as = roman_title_created_de_jure_ignore
				show_as_tooltip = {
					check_de_jure_liege_duchy_effect = yes
				}
				every_in_list = {
					variable = roman_de_jure_territory
					show_as_tooltip = {
						check_de_jure_liege_county_effect = yes
					}
				}
			}
			show_as_tooltip = {
				if = {
					limit = {
						exists = this.var:selected_title.holder
						this.var:selected_title.holder = { top_liege_can_create_roman_title_trigger = yes }
					}
					show_as_tooltip = {
						destroy_title = this.var:selected_title
					}
				}
			}
		}
		hidden_effect = {
			every_in_global_list = {
				variable = roman_duchy_list
				limit = { 
					ROOT = { 
						can_destroy_formable_title = { 
							TITLE = PREV
							COUNTY_TRIGGER = "NOT = { roman_controlled_trigger = yes }" 
						} 
					}
					has_variable = roman_title_created_de_jure
				}
				ROOT = { destroy_roman_duchy_effect = { TITLE = PREV } }
			}
			remove_variable = selected_title
		}
	}
	
	ai_potential = {
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
	
	widget = {
		gui = "decision_view_widget_destroy_roman_de_jure_duchies"
		controller = decision_option_list_controller

		### This placeholder has to be here or the game will crash
		item = {
			value = none
			is_shown = { always = yes }
			is_valid = { always = yes }
			localization = placeholder_name
		}
	}
}
