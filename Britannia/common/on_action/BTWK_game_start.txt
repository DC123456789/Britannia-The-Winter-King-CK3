BTWK_story_trigger_on_game_start_479 = {
	effect = {
		if = {
			limit = { has_game_rule = BTWK_enabled }		
			character:1 = {
				add_character_modifier = {
					modifier =  story_health_bonus_large_modifier
				}
				create_story = {
					type = story_uther_high_council	
				}
			}
		}
	}
}

			if = {
				limit = { has_variable = anglo_saxon_duchy }
				this.var:anglo_saxon_duchy = {
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
					add_to_variable_list = {
						name = de_jure_affected_territory
						target = PREV
					}
				}
				if = {
					limit = { has_variable = anglo_saxon_duchy_dependency }
					this.var:anglo_saxon_duchy_dependency = {
						add_to_variable_list = {
							name = de_jure_affected_territory
							target = PREV
						}
					}
				}
			}
			if = {
				limit = { has_variable = secondary_anglo_saxon_duchy }
				this.var:secondary_anglo_saxon_duchy = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
					add_to_variable_list = {
						name = de_jure_affected_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable = roman_duchy }
				this.var:roman_duchy = { 
					add_to_variable_list = {
						name = roman_de_jure_territory
						target = PREV
					}
					add_to_variable_list = {
						name = roman_de_jure_affected_territory
						target = PREV
					}
				}
			}
		}
		
		# Similar list for kingdoms
		every_duchy = {
			if = {
				limit = { has_variable = formable_kingdom }
				this.var:formable_kingdom = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable = expansion_kingdom }
				this.var:expansion_kingdom = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable_list = potential_integration_kingdoms }
				ordered_in_list = {
					variable = potential_integration_kingdoms
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
		}
		
		# Get a list of what duchies are under what hierarchies
		title:e_aengland = {
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_duchy }
				if = {
					limit = { has_variable_list = dynamic_de_jure_territory }
					add_to_global_variable_list = {
						name = anglo_saxon_duchy_list
						target = THIS
					}
				}
			}
		}
		every_duchy = {
			limit = { has_variable_list = roman_de_jure_territory }
			add_to_global_variable_list = {
				name = roman_duchy_list
				target = THIS
			}
		}
		
		# Initialize de jure structure
		every_county = {
			check_de_jure_liege_county_effect = yes
		}
		every_duchy = {
			check_de_jure_liege_duchy_effect = yes
		}
	}
