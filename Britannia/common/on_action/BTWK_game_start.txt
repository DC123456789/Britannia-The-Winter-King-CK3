BTWK_culture_setup = {
	effect = {
		# Set flags for parents for use in custom loc
		every_culture_global = {
			save_ancestral_non_roman_culture_effect = yes
		}
		
		# TODO Set starting cultural acceptance
		
		# Set baseline language knowledge
		# All Brythonic nobles and Christian clergy should know Latin
		# Plus "assimilating" Irish (basically Demetia)
		# Lowborn courtiers have a 75% chance as most of them are going to be minor nobility - still high 
		# status enough to be probably Latin-speaking at this point
		every_ruler = {
			limit = {
				OR = {
					culture = { has_cultural_pillar = heritage_brythonic }
					AND = {
						culture = culture:irish
						top_liege = {
							OR = {
								has_cultural_pillar = heritage_brythonic
								has_cultural_pillar = heritage_latin
								AND = {
									has_royal_court = yes
									has_court_language = language_latin
								}
								AND = {
									NOT = { has_dlc_feature = royal_court }
									has_title = title:k_demetia
								}
							}
						}
					}
				}
				# capital_province = { geographical_region = btwk_world_britannia }
			}
			if = {
				limit = { NOT = { knows_language = language_latin } }
				learn_language = language_latin
			}
			if = {
				limit = { 
					NOT = { culture = { has_cultural_pillar = heritage_brythonic } } 
					NOT = { knows_language = language_brythonic }
				}
				learn_language = language_brythonic
			}
			every_courtier_or_guest = {
				limit = {
					OR = {
						culture = { has_cultural_pillar = heritage_brythonic }
						AND = {
							NOT = { prev.culture = { has_cultural_pillar = heritage_brythonic } }
							culture = prev.culture
						}
					}
					NOT = { trait = peasant_leader }
					NOT = { this = character:10 }		# Derfel specifically doesn't know Latin
				}
				random = {
					chance = 75
					modifier = {
						add = 25
						OR = {
							is_lowborn = no
							AND = {
								is_clergy = yes
								religion = religion:christianity_religion
							}
						}
					}
					modifier = {
						add = -100
						this = character:10
					}
					if = {
						limit = { NOT = { knows_language = language_latin } }
						learn_language = language_latin
					}
					if = {
						limit = { 
							NOT = { culture = { has_cultural_pillar = heritage_brythonic } } 
							NOT = { knows_language = language_brythonic }
						}
						learn_language = language_brythonic
					}
				}
			}
		}
		
		# Custom Court Grandeur settings
		if = {
			limit = {
				has_dlc_feature = royal_court
			}
			character:1 = {		# Dumnonia
				set_current_court_grandeur = 45
			}
			character:14 = {	# Gwent
				set_current_court_grandeur = 30
			}
		}
		
		# Block some cultures in the ruler designer (mostly those that appear over the course of the game)
		every_culture_global = {
			limit = {
				OR = {
					has_cultural_pillar = heritage_romance
					has_cultural_pillar = heritage_arabic
					has_cultural_pillar = heritage_berber
					has_cultural_pillar = heritage_turkic
					AND = {
						has_cultural_pillar = heritage_west_germanic
						NOR = {
							this = culture:angle
							this = culture:old_saxon
							this = culture:jute
							this = culture:frisian
						}
					}
					AND = {
						has_cultural_pillar = heritage_frankish
						NOT = {
							this = culture:frankish
						}
					}
					AND = {
						has_cultural_pillar = heritage_brythonic
						NOR = {
							this = culture:briton
							this = culture:armorican
						}
					}
					this = culture:norse_gael
				}
			}
			set_variable = block_start_with
		}
		
		# Set global variables references for loc purposes
		set_global_variable = { name = briton_culture value = culture:briton }
		set_global_variable = { name = welsh_culture value = culture:welsh }
		set_global_variable = { name = cornish_culture value = culture:cornish }
		set_global_variable = { name = cumbrian_culture value = culture:cumbrian }
		
		set_global_variable = { name = angle_culture value = culture:angle }
		set_global_variable = { name = old_saxon_culture value = culture:old_saxon }
		set_global_variable = { name = jute_culture value = culture:jute }
		
		set_global_variable = { name = romanobritish_culture value = culture:romanobritish }
	}
}

BTWK_religion_setup = {
	effect = {		
		# Block all irrelevant vanilla faiths permanently
		every_religion_global = {
			limit = {
				NOR = {
					this = religion:basque_religion
					this = religion:celtic_religion
					this = religion:christianity_religion
					this = religion:dualism_religion
					this = religion:germanic_religion
					this = religion:hellenism_religion
					this = religion:islam_religion
					this = religion:judaism_religion
					this = religion:paganism_religion
					this = religion:scythian_religion
				}
			}
			every_faith = { set_variable = { name = block_conversion } }
		}
		religion:christianity_religion = {
			every_faith = {
				limit = {
					NOR = {
						this = faith:catholic
						this = faith:pelagian
						this = faith:arian
						this = faith:priscillianism
						this = faith:novatianist
						this = faith:adoptionist
						this = faith:valentinianism
					}
				}
				set_variable = { name = block_conversion } 
			}
		}
		religion:dualism_religion = {
			every_faith = {
				limit = {
					NOR = {
						this = faith:manichean
					}
				}
				set_variable = { name = block_conversion } 
			}
		}
		religion:islam_religion = {
			every_faith = {
				limit = {
					NOR = {
						this = faith:ashari
						this = faith:muwalladi
						this = faith:maturidi
						this = faith:mutazila
						this = faith:ismaili
						this = faith:zayidi
						this = faith:ibadi
						this = faith:sufri
						this = faith:barghawata
					}
				}
				set_variable = { name = block_conversion } 
			}
		}
		religion:judaism_religion = {
			every_faith = {
				limit = {
					NOR = {
						this = faith:rabbinism
					}
				}
				set_variable = { name = block_conversion } 
			}
		}
		
		# Block all relevant Muslim faiths until it appears
		religion:islam_religion = {
			every_faith = {
				limit = {
					NOT = { has_variable = block_conversion }
				}
				set_variable = { name = block_conversion_till_founded }
			}
		}
	}
}
