		every_county = {
			if = {
				limit = { has_variable = anglo_saxon_duchy }
				this.var:anglo_saxon_duchy = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable = secondary_anglo_saxon_duchy }
				this.var:secondary_anglo_saxon_duchy = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable = roman_duchy }
				this.var:roman_duchy = { 
					add_to_variable_list = {
						name = roman_de_jure_territory
						target = PREV
					}
				}
			}
		}
		
		# Similar list for kingdoms
		every_duchy = {
			if = {
				limit = { has_variable = formable_kingdom }
				this.var:formable_kingdom = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable = expansion_kingdom }
				this.var:expansion_kingdom = { 
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
			if = {
				limit = { has_variable_list = potential_integration_kingdoms }
				ordered_in_list = {
					variable = potential_integration_kingdoms
					add_to_variable_list = {
						name = dynamic_de_jure_territory
						target = PREV
					}
				}
			}
		}
