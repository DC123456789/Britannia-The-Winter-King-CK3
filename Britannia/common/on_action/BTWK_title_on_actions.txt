# A title is destroyed
# root = the holder before destruction
# scope:landed_title = the title that is being destroyed
# Does *not* provide scope:transfer_type
on_title_destroyed = {
	on_actions = {
		BTWK_on_title_destroyed
	}
}

BTWK_on_title_destroyed = {
	effect = {
		if = {
			limit = {
				scope:landed_title = title:e_britannia
				BTWK_britannia_high_kingship_active_trigger = yes
			}
			# Apply penalty to dynastic High King Authority
			dynasty = {
				if = {
					limit = { has_variable = britannia_high_kingship_authority }
					change_variable = {
						name = britannia_high_kingship_authority
						multiply = BTWK_britannia_high_king_authority_inherited_current_proportion
					}
				}
			}
			
			# Revoke all reforms + effects if any were passed
			if = {
				limit = { 
					BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = formalize_high_kingship }
					BTWK_britannia_high_kingship_active_trigger = yes
				}
				BTWK_britannia_high_king_revoke_all_reforms_effect = yes
			}
		}
		
		# Check for Fall of Anglia event
		if = {
			limit = {
				scope:landed_title = title:k_east_anglia
				title:k_east_anglia = { has_variable = anglia_post_aelle_crisis }
			}
			save_scope_as = former_king
			scope:former_king = { trigger_event = BTWK_anglo_saxon_invasions.0011 }
			every_player = {
				limit = { this != scope:former_king }
				trigger_event = BTWK_anglo_saxon_invasions.0011
			}
		}
	}
}

# A title is usurped by a character
# root = the new holder
# scope:title = the title that changes hands
# scope:previous_holder = previous holder. Shouldn't be dead
on_title_gain = {
	on_actions = {
		BTWK_on_title_gain
		delay = { days = 1 }
		BTWK_on_title_gain_delayed
	}
}

BTWK_on_title_gain = {
	events = {
		titular_titles_anglosaxon.0100 	# Anglo-Saxon Kingdoms
		titular_titles_roman.0003 		# Roman Provinces
		titular_titles_briton.0007 		# High Kingdom of Yr Hen Ogledd
		titular_titles_briton.0012 		# High Kingdom of Cymru
		titular_titles_irish.0005		# High Kingdom of Comruig (Irish Cymru)
		
		delay = { days = 1 }			# This makes sure that the player actually sees the event in inheritances
		spreading_of_christianity.0100		# First Anglo-Saxon Christian King
	}
	effect = {
		if = {
			limit = {
				has_variable = newly_landed_vassal
				government_has_flag = government_has_tribal_obligations_contract
			}
			vassal_contract_set_obligation_level = { type = tribal_obligations level = 0 }
		}
		
		# Update cultural redoubts
		if = {
			limit = { is_independent_ruler = yes }
			BTWK_update_cultural_redoubt_modifier_for_sub_realm_effect = yes
		}
		
		# Update Britannia Mint/Coinage modifer
		if = {
			limit = { BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = establish_central_mint } }
			BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
			if = {
				limit = { scope:title = title:e_britannia }
				# If the High Kingship title itself changed hands, check everything
				scope:previous_holder = {
					BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
					every_vassal_or_below = {
						BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
					}
					every_tributary = {
						limit = {
							BTWK_britannia_high_kingship_is_subject_king_tributary_trigger = yes
						}
						BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
						every_vassal_or_below = {
							BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
						}
					}
				}
				every_vassal_or_below = {
					BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
				}
				every_tributary = {
					limit = {
						BTWK_britannia_high_kingship_is_subject_king_tributary_trigger = yes
					}
					BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
					every_vassal_or_below = {
						BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
					}
				}				
			}
		}
		
		# Update Britannia State Church Modifier
		if = {
			limit = {
				BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = centralize_the_church }
				has_title = title:e_britannia
			}
			BTWK_britannia_high_king_reform_centralize_the_church_check_modifier_effect = yes
		}
		
		# Transfer High Kingship story once the High Kingship becomes inheritable
		if = {
			limit = {
				scope:title = title:e_britannia
				BTWK_britannia_high_kingship_active_trigger = yes
				BTWK_britannia_high_kingship_can_inherit_trigger = yes
			}
			BTWK_britannia_initialize_high_kingship_effect = yes
		}
		
		# Create Bretwalda story
		if = {
			limit = {
				scope:title = title:e_aengland
				BTWK_aengland_bretwalda_active_trigger = yes
			}
			create_story = BTWK_aengland_bretwalda_story
		}
		
		# Transfer Rise of Anglo-Saxon Kingdom story
		if = {
			limit = { 
				scope:previous_holder ?= { 
					owns_story_of_type = BTWK_rise_of_new_anglo_saxon_kingdom_story
					OR = {
						is_alive = no
						scope:title.tier > highest_held_title_tier
					}
				}
			}
			save_scope_as = new_holder
			scope:previous_holder ?= { 
				every_owned_story = {
					limit = { story_type = BTWK_rise_of_new_anglo_saxon_kingdom_story }
					make_story_owner = scope:new_holder
				}
			}
		}
		
		# Trigger Fall of Aelle event
		if = {
			limit = { scope:title = title:k_east_anglia }
			if = {
				limit = { title:k_east_anglia = { has_variable = aelle_ruling_anglia } }
				character:80 = { save_scope_as = aelle_of_anglia }
				save_scope_as = new_holder
				scope:new_holder = { trigger_event = BTWK_anglo_saxon_invasions.0010 }
				every_player = {
					limit = { this != scope:new_holder }
					trigger_event = BTWK_anglo_saxon_invasions.0010
				}
			}
			# Pass on modifier if the crisis is ongoing
			else_if = {
				limit = { title:k_east_anglia = { has_variable = anglia_post_aelle_crisis } }
				scope:new_holder = { add_character_modifier = BTWK_anglia_unstable_realm }
				scope:previous_holder ?= { remove_character_modifier = BTWK_anglia_unstable_realm }
			}
		}
	}
}

BTWK_on_title_gain_delayed = {
	effect = {		
		# Make sure the High King is always properly initialized - if the High Kingship is not yet
		# inheritable, we do this one day later to make sure that we don't accidentally fire this for
		# the heir who will have the title immediately destroyed
		if = {
			limit = {
				scope:title = title:e_britannia
				has_title = title:e_britannia
				BTWK_britannia_high_kingship_active_trigger = yes
				BTWK_britannia_high_kingship_can_inherit_trigger = no
			}
			BTWK_britannia_initialize_high_kingship_effect = yes
		}
	}
}

# A title is usurped by a character
# root = the new holder
# scope:title = the title that changes hands
# scope:previous_holder = previous holder. Shouldn't be dead
on_title_gain_usurpation = {
	on_actions = {
		BTWK_on_title_gain_usurpation
		delay = { days = 1 }
		BTWK_on_title_gain_delayed
	}
}

BTWK_on_title_gain_usurpation = {
	effect = {
		if = {
			limit = { # FP2
				# We are only interested in the title within the Struggle Region
				title_capital_county.title_province = { geographical_region = btwk_world_britannia }
				# Only for usurpation
				scope:transfer_type = flag:usurped

				root = {
					any_character_struggle = {
						involvement = involved
						activate_struggle_catalyst_secondary_character_involvement_involved_trigger = {
							CATALYST = catalyst_usurp_title_britannia_region
							CHAR = root
						}
					}
				}
			}
			root = {
				every_character_struggle = {
					involvement = involved
					limit = {
						activate_struggle_catalyst_secondary_character_involvement_involved_trigger = {
							CATALYST = catalyst_usurp_title_britannia_region
							CHAR = root
						}
					}
					activate_struggle_catalyst = {
						catalyst = catalyst_usurp_title_britannia_region
						character = root
					}
				}
			}
		}
	}
}

# A title is lost by a character
# root = the old holder
# scope:title = the title that changes hands
# scope:new_holder = new holder.
on_title_lost = {
	on_actions = {
		BTWK_on_title_lost
	}
}

# Note - Isn't triggered when a title gets destroyed!
BTWK_on_title_lost = {
	effect = {
		if = {
			limit = {
				is_playable_character = no
			}
			remove_tribal_fame_effect = yes
			
			# End Rise of Anglo-Saxon Kingdom story
			if = {
				limit = { owns_story_of_type = BTWK_rise_of_new_anglo_saxon_kingdom_story }
				every_owned_story = {
					limit = { story_type = BTWK_rise_of_new_anglo_saxon_kingdom_story }
					set_variable = rise_of_anglo_saxon_kingdom_story_defeated
					end_story = yes
				}
			}
		}
		if = {
			limit = {
				scope:title = title:e_britannia
				BTWK_britannia_high_kingship_active_trigger = yes
			}
			# Destroy High Kingship on succession
			if = {
				limit = { 
					owns_story_of_type = BTWK_britannia_high_kingship_story 
					BTWK_britannia_high_kingship_can_inherit_trigger = no
				}
				save_scope_as = former_high_king
				set_variable = { name = player_heir_char value = this.player_heir }
				every_owned_story = {
					limit = { story_type = BTWK_britannia_high_kingship_story }
					set_variable = high_king_abdicated
					end_story = yes
				}
			}
			# Apply penalty to dynastic High King Authority
			dynasty = {
				if = {
					limit = { 
						has_variable = britannia_high_kingship_authority
						NOT = { BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = formalize_succession } }
					}
					change_variable = {
						name = britannia_high_kingship_authority
						multiply = BTWK_britannia_high_king_authority_inherited_proportion
					}
				}
			}
			# Other cleanup
			remove_variable = BTWK_britannia_high_king_reform_trying_to_pass
		}
		
		if = {
			limit = {
				scope:title = title:e_aengland
				BTWK_aengland_bretwalda_active_trigger = yes
			}
			# Destroy Bretwalda on succession
			if = {
				limit = { 
					BTWK_aengland_bretwalda_can_inherit_trigger = no
				}
				save_scope_as = former_bretwalda
				player_heir = { save_scope_as = former_bretwalda_player_heir }
				title:e_aengland.holder ?= {
					destroy_title = title:e_aengland
					save_scope_as = former_bretwalda_player_heir
					save_scope_as = background_throne_room_scope
				}
				trigger_event = { id = BTWK_aengland_bretwalda.0021 }
				every_player = {
					limit = {
						NOT = { prev ?= this }
						BTWK_aengland_bretwalda_is_relevant_trigger = yes
					}
					trigger_event = BTWK_aengland_bretwalda.0021
				}
			}
			# Otherwise, apply penalty to and inherit hegemony score
			else = {
				if = {
					limit = { 
						has_variable = aengland_bretwalda_hegemony_score
					}
					change_variable = {
						name = aengland_bretwalda_hegemony_score
						multiply = BTWK_britannia_high_king_authority_inherited_proportion
					}
					title:e_aengland.holder = {
						change_variable = {
							name = aengland_bretwalda_hegemony_score
							add = prev.var:aengland_bretwalda_hegemony_score
						}
					}
				}
			}
			# Cleanup
			every_owned_story = {
				limit = { story_type = BTWK_aengland_bretwalda_story }
				end_story = yes
			}
		}
		
		# Update Britannia Mint/Coinage modifer
		if = {
			limit = { BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = establish_central_mint } }
			BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
		}
		
		# Update Britannia State Church Modifier
		if = {
			limit = {
				BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = centralize_the_church }
				OR = {
					has_title = title:e_britannia
					scope:title = title:e_britannia
				}
			}
			BTWK_britannia_high_king_reform_centralize_the_church_check_modifier_effect = yes
		}
		
		if = {
			limit = {
				is_playable_character = no
			}
			
			# Clear Aengland Hegemony Score - happens at end, after event triggers
			if = {
				limit = { has_variable = aengland_bretwalda_hegemony_score }
				set_variable = {
					name = aengland_bretwalda_hegemony_score
					value = 0
				}
			}
		}
	}
}

# A claim is gained by a character
# root = the claimant
# scope:title = the title that is claimed
# scope:transfer_type = flag:inheritance or none

	effect = {

			limit = {

				root = {
					any_character_struggle = {
						activate_struggle_catalyst_secondary_character_involvement_involved_trigger = {
							CATALYST = catalyst_gain_claim_on_title_britannia_region
							CHAR = root
						}
					}
				}
				scope:title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
			root = {
				every_character_struggle = {
					involvement = involved
					limit = {
						activate_struggle_catalyst_secondary_character_involvement_involved_trigger = {
							CATALYST = catalyst_gain_claim_on_title_britannia_region
							CHAR = root
						}
					}
					activate_struggle_catalyst = {
						catalyst = catalyst_gain_claim_on_title_britannia_region
						character = root
					}
				}
			}
		}
	}
}

# A title change makes the character rank up in the landed tiers (eg Duke -> King)
# root = character ranking up
# scope:title = old primary title
on_rank_up = { # Will not fire during history execution or for dying characters
	on_actions = {
		btwk_on_rank_up
	}
}

btwk_on_rank_up = {
	effect = {
		# Courtiers getting granted titles by tribal rulers granting titles should 
		# remain tribal instead of becoming feudal, even if the holding has a castle
		if = {
			limit = {
				scope:transfer_type = flag:granted
				NOT = { exists = scope:title }
				government_has_flag = government_is_feudal
				primary_title.previous_holder ?= { government_has_flag = government_is_tribal }
			}
			set_variable = changed_to_tribal
			change_government = primary_title.previous_holder.government_type
		}
		
		# Turn king-level vassals of Britannia into subject kingdoms before the Empower the High Council reform
		BTWK_britannia_high_kingship_check_king_tier_vassals_effect = yes
		# Turn king-level vassals of Aengland into tributaries before Declaring Supremacy
		BTWK_aengland_bretwalda_check_king_tier_vassals_effect = yes
	}
}

# A vassal changes their liege
# root = vassal changing their status
# scope:old_liege = the previous liege of the vassal
# It is also fired for rulers gaining independence, those can be identified by checking current liege
on_vassal_change = {
	on_actions = {
		btwk_on_vassal_change
	}
}

btwk_on_vassal_change = {
	events = {
		spreading_of_christianity.0100		# First Anglo-Saxon Christian King
	}
	effect = {
		# Update cultural redoubts
		BTWK_update_cultural_redoubt_modifier_for_sub_realm_effect = yes
		
		# Turn king-level vassals of Britannia into subject kingdoms before the Empower the High Council reform
		BTWK_britannia_high_kingship_check_king_tier_vassals_effect = yes
		
		# Turn king-level vassals of Aengland into tributaries before Declaring Supremacy
		BTWK_aengland_bretwalda_check_king_tier_vassals_effect = yes

		# Update Britannia Mint/Coinage modifer
		if = {
			limit = { BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = establish_central_mint } }
			BTWK_britannia_high_king_reform_establish_central_mint_check_modifier_effect = yes
		}
		
		# Update Britannia State Church Modifier
		if = {
			limit = { BTWK_britannia_high_king_reform_is_passed_trigger = { REFORM_NAME = centralize_the_church } }
			every_liege_or_above = {
				limit = {
					has_title = title:e_britannia
				}
				BTWK_britannia_high_king_reform_centralize_the_church_check_modifier_effect = yes
			}
			scope:old_liege = {
				if = {
					limit = {
						has_title = title:e_britannia
					}
					BTWK_britannia_high_king_reform_centralize_the_church_check_modifier_effect = yes
				}
			}
		}
	}
}
