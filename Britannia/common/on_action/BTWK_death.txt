on_death = {
	on_actions = {BTWK_uther_dies_on_action}
}


#Trigger to return Arthur to Dumnonia a few weeks after Uther's death (simulate time for news to reach Arthur, & for his return)

BTWK_uther_dies_on_action = {
	effect = {
		if = {
			limit = {
				root = character:1
				NOT = { has_global_variable = uther_is_dead }
				has_game_rule = BTWK_enabled
			}	
			character:3 = {
				trigger_event = {
					id = BTWK_story_event.0108
				}
			}
		}
	}
}
