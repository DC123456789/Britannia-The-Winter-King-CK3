# character just about to die in root scope
# if a killer is known, it's set as scope:killer
#Triggered by code
on_death = {
	on_actions = {
		BTWK_on_death
	}
}

BTWK_on_death = {
	effect = {
		# Britannia High Kingship Cleanup
		every_in_list = {
			variable = britannia_potential_supporter_kings_linked_list
			end_story = yes
		}
		clear_variable_list = britannia_potential_supporter_kings_linked_list
		every_in_list = {
			variable = britannia_potential_opponent_kings_linked_list
			end_story = yes
		}
		clear_variable_list = britannia_potential_opponent_kings_linked_list
		every_in_list = {
			variable = britannia_potential_neutral_kings_linked_list
			end_story = yes
		}
		clear_variable_list = britannia_potential_neutral_kings_linked_list
		if = {
			limit = { 
				this = character:1 
				has_variable = has_BTWK_story_uther_high_council_story
			}
			character:3 = {
				trigger_event = {
					id = BTWK_story_event.0112
				}
			}
		}
		
		# Handle High King death events
		if = {
			limit = { 
				owns_story_of_type = BTWK_britannia_claim_high_kingship_story
			}
			save_scope_as = claimant
			set_variable = BTWK_britannia_high_king_dying_claimant
			set_variable = BTWK_britannia_high_king_was_supporter
			trigger_event = BTWK_britannia_high_king.0013
		}
		if = {
			limit = { 
				owns_story_of_type = BTWK_britannia_high_kingship_story
				BTWK_britannia_high_kingship_can_inherit_trigger = no
			}
			save_scope_as = former_high_king
			trigger_event = BTWK_britannia_high_king.0023
		}
		
		# Handle Bretwalda death events
		if = {
			limit = { 
				has_title = title:e_aengland
				BTWK_aengland_bretwalda_can_inherit_trigger = no
			}
			save_scope_as = former_bretwalda
			title:e_aengland.current_heir = { save_scope_as = former_bretwalda_player_heir }
			scope:former_bretwalda_player_heir = { trigger_event = BTWK_aengland_bretwalda.0020 }
			every_player = {
				limit = { 
					NOR = {
						scope:former_bretwalda ?= this
						scope:former_bretwalda.player_heir ?= this 
					}
					BTWK_aengland_bretwalda_is_relevant_trigger = yes
				}
				trigger_event = BTWK_aengland_bretwalda.0020
			}
		}
		
		# Trigger Death of Aelle event
		if = {
			limit = { 
				this = character:80
				has_title = title:k_east_anglia
				title:k_east_anglia = { 
					has_variable = aelle_ruling_anglia
					NOT = { has_variable = triggered_aelle_fall_event }
				}
			}
			title:k_east_anglia = { set_variable = triggered_aelle_fall_event }
			character:80 = { save_scope_as = aelle_of_anglia }
			title:k_east_anglia.current_heir = { save_scope_as = new_holder }
			scope:new_holder = { trigger_event = BTWK_anglo_saxon_invasions.0010 }
			every_player = {
				limit = { this != scope:new_holder }
				trigger_event = BTWK_anglo_saxon_invasions.0010
			}
		}
		
		# Struggle catalysts
		## High King deaths.
		if = {
			limit = { has_title = title:e_britannia }
			## Did they die violently?
			### Checked against the struggle directly
			if = {
				limit = {
					exists = scope:killer
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_high_king_britannia_died_violently
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_high_king_britannia_died_violently
						character = root
					}
					# We set the flag at the very end because we *do* want it 
					# to potentially stack for the same character
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
			## If not, give the natural version.
			### Checked against the struggle directly
			else_if = {
				limit = {
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_high_king_britannia_died_naturally
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_high_king_britannia_died_naturally
						character = root
					}
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
		}
		## Bretwalda deaths.
		if = {
			limit = { has_title = title:e_aengland }
			## Did they die violently?
			### Checked against the struggle directly
			if = {
				limit = {
					exists = scope:killer
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_bretwalda_died_violently
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_bretwalda_died_violently
						character = root
					}
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
			## If not, give the natural version.
			### Checked against the struggle directly
			else_if = {
				limit = {
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_bretwalda_died_naturally
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_bretwalda_died_naturally
						character = root
					}
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
		}
		## Conquerer deaths - may stack with one of the above!
		if = {
			limit = { 
				owns_story_of_type = story_conqueror
				primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
			## Did they die violently?
			### Checked against the struggle directly
			if = {
				limit = {
					exists = scope:killer
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_conqueror_died_violently
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_conqueror_died_violently
						character = root
					}
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
			## If not, give the natural version.
			### Checked against the struggle directly
			else_if = {
				limit = {
					struggle:britannia_struggle ?= {
						phase_has_catalyst = catalyst_conqueror_died_naturally
						NOT = { has_variable = significant_character_death_cooldown }
					}
				}
				struggle:britannia_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_conqueror_died_naturally
						character = root
					}
					save_scope_as = flag_significant_character_death_cooldown
				}
			}
		}
		# We set the flag at the very end because we *do* want it to potentially stack for the same character
		if = {
			limit = { scope:flag_significant_character_death_cooldown ?= struggle:britannia_struggle }
			struggle:britannia_struggle = {
				set_variable = {
					name = significant_character_death_cooldown
					value = yes
					years = catalyst_cooldown_between_significant_character_deaths_years_value
				}
			}
		}
	}
}
