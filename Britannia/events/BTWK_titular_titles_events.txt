# Annex the North Bank (Mercia)
titular_titles_anglosaxon.0004 = {
	type = character_event
	content_source = dlc_BTWK
	hidden = yes
	
	immediate = {
		save_scope_as = conqueror
		trigger_event = { id = titular_titles_anglosaxon.0005 }
		every_player = {
			limit = { NOT = { this = PREV } }
			trigger_event = { id = titular_titles_anglosaxon.0005 }
		}
		title:k_mercia = { set_variable = annexed_north_bank }
		# This is where the actual de jure shift happens
		# It happens immediately after the notification events are fired,
		# so that the correct tooltips are generated but the de jure shift 
		# has already occured by the time the players actually see it
		hidden_effect = {
			title:d_cilternsaete = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			title:d_south_angles = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			title:d_hwicce = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			if = {
				limit = {
					title:d_berkshire.title_capital_county = title:c_dorchester
				}
				title:d_berkshire = {
					set_variable = custom_capital
					set_capital_county = title:c_vedonium
				}
			}
			if = {
				limit = {
					title:k_wessex.title_capital_county = title:c_dorchester
				}
				title:k_wessex = {
					set_variable = custom_capital
					set_capital_county = title:c_venta
				}
			}
		}
	}
}

# Creation of kingdom
titular_titles_anglosaxon.0100 = {
	type = character_event
	content_source = dlc_BTWK
	hidden = yes
	
	trigger = {
		# scope:transfer_type = flag:created
		scope:title = {
			is_anglo_saxon_kingdom_trigger = yes
			NOT = { has_variable = title_has_been_created }
		}
	}
	
	immediate = {
		save_scope_as = founder
		hidden_effect = {
			# Rename Jutish Wessex to Westeotas ("West Jutes")
			# Note that this can also happen if the cultural divergence event hits Wessex
			# before it forms and Jutish culture is dominant in the area
			if = {
				limit = {
					scope:title = title:k_wessex
					scope:founder = { is_culture_or_descendant_trigger = { CULTURE = culture:jute } }
				}
				title:k_wessex = {
					set_title_name = WESTEOTAS
					set_capital_county = title:c_clausentum
					set_variable = jutish_wessex
				}
			}
			# Rename Wales to Westerne if it is created by a "Westernian" culture character,
			# or if it is the creator's first kingdom title
			if = {
				limit = { 
					NOT = { has_variable = title_has_been_created }
					scope:founder = {
						OR = {
							culture.var:anglo_saxon_regional_culture_kingdom ?= title:k_wales
							AND = {
								is_independent_ruler = yes
								highest_held_title_tier = tier_kingdom
								NOT = { 
									any_held_title = {
										count >= 2
										tier >= tier_kingdom
									}
								}
							}
						}
					}
				}
				title:k_wales = {
					set_title_name = WESTERNE
					set_variable = wales_renamed_westerne
				}
			}
		}
		trigger_event = { id = titular_titles_anglosaxon.0101 }
		every_player = {
			limit = { NOT = { this = PREV } }
			trigger_event = { id = titular_titles_anglosaxon.0101 }
		}
		scope:title = { set_variable = title_has_been_created }
	}
}

	scope = none

		if = {

				limit = { NOT = { has_global_variable = end_of_britannia_timer_started } }

				set_global_variable = { name = end_of_britannia_timer days = 3650 }

					NOT = { has_global_variable = end_of_britannia_timer }

				remove_global_variable = end_of_britannia_timer_started
				remove_global_variable = end_of_britannia_timer
				trigger_event = { id = titular_titles_briton.0051 }
			}
		}
		else = {
			if = {
				limit = { has_global_variable = end_of_britannia_timer_started }
				remove_global_variable = end_of_britannia_timer_started
				remove_global_variable = end_of_britannia_timer

				save_scope_as = britannia_cultural_kingdom_title
			}
		}
		title:e_romanized_germanic_britannia = {
			set_variable = { 
				name = romanized_germanic_empire_culture
				value = scope:founder.culture.var:ancestral_non_roman_culture 
			}
			set_title_name_w_tooltip_effect = { NEW_NAME = e_romanized_germanic_britannia_FORMED }
			# COA is set in common/coat_of_arms/dynamic_definitions/BTWK_e_romanized_germanic_britannia_coas
			# to handle religion switches better
			set_capital_county_w_tooltip_effect = { CAPITAL = scope:britannia_cultural_kingdom_title.title_capital_county }
		}
		trigger_event = { id = titular_titles_romanized_germanic.0004 }
		every_player = {
			limit = { NOT = { this = PREV } }
			trigger_event = { id = titular_titles_romanized_germanic.0004 }
		}
		# This is where the actual de jure shift happens
		# It happens immediately after the notification events are fired,
		# so that the correct tooltips are generated but the de jure shift 
		# has already occured by the time the players actually see it
		title:e_romanized_germanic_britannia = {
			every_in_list = {
				variable = base_kingdom_regions
				set_kingdom_region_de_jure_liege_effect = {
					REGION_BASE = THIS
					NEW_DE_JURE_LIEGE = prev
				}
			}
		}
		hidden_effect = {
			scope:britannia_cultural_kingdom_title = {
				rename_britannia_cultural_kingdom_title_effect = yes
			}
		}
	}
}
