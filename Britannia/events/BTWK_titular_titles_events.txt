# Annex the North Bank (Mercia)
titular_titles_anglosaxon.0004 = {
	type = character_event
	hidden = yes
	
	immediate = {
		save_scope_as = conqueror
		trigger_event = { id = titular_titles_anglosaxon.0005 }
		every_player = {
			limit = { NOT = { this = PREV } }
			trigger_event = { id = titular_titles_anglosaxon.0005 }
		}
		title:k_mercia = { set_variable = annexed_north_bank }
		# This is where the actual de jure shift happens
		# It happens immediately after the notification events are fired,
		# so that the correct tooltips are generated but the de jure shift 
		# has already occured by the time the players actually see it
		hidden_effect = {
			title:d_cilternsaete = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			title:d_south_angles = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			title:d_hwicce = { 
				set_variable = title_expanded_de_jure
				every_in_list = {
					variable = dynamic_de_jure_territory
					check_de_jure_liege_effect = yes
				}
			}
			if = {
				limit = {
					title:d_berkshire.title_capital_county = title:c_dorchester
				}
				title:d_berkshire = {
					set_variable = custom_capital
					set_capital_county = title:c_vedonium
				}
			}
			if = {
				limit = {
					title:k_wessex.title_capital_county = title:c_dorchester
				}
				title:k_wessex = {
					set_variable = custom_capital
					set_capital_county = title:c_venta
				}
			}
		}
	}
}

titular_titles_roman.0001 = {
	type = character_event
	title = titular_titles_roman.0001.t
	desc = {
		desc = titular_titles_roman.0001.desc.intro
		first_valid = {
			desc = titular_titles_roman.0001.desc.fallback
		}
		desc = titular_titles_roman.0001.desc.outro
	}
	theme = realm
	override_background = terrain
	
	left_portrait = {
		character = scope:conqueror
		animation = personality_honorable
	}
	
	immediate = {
		if = {
			limit = { this = scope:conqueror }
			if = {
				limit = { 
					exists = scope:new_roman_duchy_title.holder 
					NOR = {
						scope:new_roman_duchy_title.holder = ROOT
						scope:new_roman_duchy_title.holder = { is_vassal_or_below_of = ROOT }
					}
				}
				create_title_and_vassal_change = {
					type = usurped
					save_scope_as = title_change
					add_claim_on_loss = yes
				}
				scope:new_roman_duchy_title = {
					change_title_holder = {
						holder = root
						change = scope:title_change
					}
				}
				resolve_title_and_vassal_change = scope:title_change
			}
			else_if = {
				limit = { NOT = { exists = scope:new_roman_duchy_title.holder } }
				create_title_and_vassal_change = {
					type = created
					save_scope_as = title_change
					add_claim_on_loss = no
				}
				scope:new_roman_duchy_title = {
					change_title_holder = {
						holder = root
						change = scope:title_change
					}
				}
				resolve_title_and_vassal_change = scope:title_change
			}
		}
		every_in_global_list = {
			variable = celtic_duchy_list
			if = {
				limit = {
					is_capital_in_roman_duchy_trigger = { DUCHY = scope:new_roman_duchy_title }
					NOT = { has_variable_list = roman_de_jure_territory }
				}
				if = {
					limit = { exists = holder }
					# destroy_title must be in character scope to work
					holder = { destroy_title = prev }
				}
			}
		}
		show_as_tooltip = {
			scope:new_roman_duchy_title = {
				check_de_jure_liege_effect = yes
				every_in_list = {
					variable = roman_de_jure_affected_territory
					check_de_jure_liege_effect = yes
				}
			}
		}
	}

	option = {
		name = {
			trigger = { 
				culture = { has_cultural_pillar = heritage_latin }
			}
			text = titular_titles_roman.0001.a.roman
		}
		name = {
			trigger = { 
				culture = { has_cultural_pillar = heritage_brythonic }
			}
			text = titular_titles_roman.0001.a.briton
		}
		name = {
			trigger = { 
				culture = {
					has_cultural_pillar = heritage_west_germanic
				}
			}
			text = titular_titles_roman.0001.a.anglosaxon
		}
		name = {
			text = titular_titles_roman.0001.a
		}
	}
}

titular_titles_briton.0002 = {
	type = character_event
	title = titular_titles_briton.0002.t
	desc = titular_titles_briton.0002.desc
	theme = realm
	
	left_portrait = {
		character = scope:founder
		animation = personality_honorable
	}
	
	immediate = {
		if = {
			limit = { this = scope:founder }
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = no
			}
			title:k_deheubarth = {
				change_title_holder = {
					holder = root
					change = scope:title_change
				}
				copy_title_history = title:k_demetia
			}
			resolve_title_and_vassal_change = scope:title_change
			destroy_title = title:k_demetia
			title:k_demetia = {
				hidden_effect = {
					every_claimant = {
						if = {
							limit = { has_strong_claim_on = title:k_demetia }
							remove_claim = title:k_demetia
							add_pressed_claim = title:k_deheubarth
						}
						else = {
							remove_claim = title:k_demetia
							add_unpressed_claim = title:k_deheubarth
						}
					}
				}
			}
		}
		# This shows the de jure shift as a tooltip
		show_as_tooltip = {
			title:k_deheubarth = { 
				save_scope_as = title_created_de_jure_pretend
				every_in_list = {
					variable = de_jure_affected_territory
					show_as_tooltip = {
						check_de_jure_liege_effect = yes
					}
