# Handle yearly tick catalysts
# Complements but does not replace/override neutral_struggle.0001
BTWK_britannia_struggle.0001 = {
	hidden = yes
	scope = struggle

	immediate = {
		save_scope_as = current_struggle
		
		# Other time-based catalysts.
		## Britannia High Kingship Reform Status
		if = {
			limit = {
				BTWK_britannia_high_kingship_can_inherit_trigger = yes
				exists = title:e_britannia.holder
			}
			if = {
				limit = { phase_has_catalyst = catalyst_high_king_britannia_is_inheritable }
				activate_struggle_catalyst = {
					catalyst = catalyst_high_king_britannia_is_inheritable
					character = title:e_britannia.holder
				}
			}
		}
		if = {
			limit = {
				BTWK_britannia_subject_king_vassals_active_trigger = yes
				exists = title:e_britannia.holder
			}
			if = {
				limit = { phase_has_catalyst = catalyst_high_king_britannia_vassalized_subject_kings }
				activate_struggle_catalyst = {
					catalyst = catalyst_high_king_britannia_vassalized_subject_kings
					character = title:e_britannia.holder
				}
			}
		}
		# Aengland Centralization Status
		# TODO
		# Britannia realm sizes
		every_independent_ruler = {
			limit = {
				primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
			if = {
				limit = { 
					realm_size >= 130
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_6 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_6
						character = prev
					}
				}
			}
			else_if = {
				limit = { 
					realm_size >= 100
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_5 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_5
						character = prev
					}
				}
			}
			else_if = {
				limit = { 
					realm_size >= 80
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_4 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_4
						character = prev
					}
				}
			}
			else_if = {
				limit = { 
					realm_size >= 60
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_3 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_3
						character = prev
					}
				}
			}
			else_if = {
				limit = { 
					realm_size >= 45
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_2 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_2
						character = prev
					}
				}
			}
			else_if = {
				limit = { 
					realm_size >= 30
					scope:current_struggle = { phase_has_catalyst = catalyst_large_realm_in_britannia_1 }
				}
				scope:current_struggle = {
					activate_struggle_catalyst = {
						catalyst = catalyst_large_realm_in_britannia_1
						character = prev
					}
				}
			}
		}
		# Britannia uninvolved control
		save_scope_value_as = {
			name = britannia_uninvolved_control
			value = {
				value = 0
				every_involved_county = {
					limit = { holder.top_liege = { NOT = { any_character_struggle = { this = scope:current_struggle } } } }
					add = 1
				}
			}
		}
		if = {
			limit = { 
				scope:britannia_uninvolved_control >= 80
				phase_has_catalyst = catalyst_uninvolved_presence_in_britannia_5
			}
			activate_struggle_catalyst = catalyst_uninvolved_presence_in_britannia_5
		}
		else_if = {
			limit = { 
				scope:britannia_uninvolved_control >= 40
				phase_has_catalyst = catalyst_uninvolved_presence_in_britannia_4
			}
			activate_struggle_catalyst = catalyst_uninvolved_presence_in_britannia_4
		}
		else_if = {
			limit = { 
				scope:britannia_uninvolved_control >= 20
				phase_has_catalyst = catalyst_uninvolved_presence_in_britannia_3
			}
			activate_struggle_catalyst = catalyst_uninvolved_presence_in_britannia_3
		}
		else_if = {
			limit = { 
				scope:britannia_uninvolved_control >= 10
				phase_has_catalyst = catalyst_uninvolved_presence_in_britannia_2
			}
			activate_struggle_catalyst = catalyst_uninvolved_presence_in_britannia_2
		}
		else_if = {
			limit = { 
				scope:britannia_uninvolved_control >= 5
				phase_has_catalyst = catalyst_uninvolved_presence_in_britannia_1
			}
			activate_struggle_catalyst = catalyst_uninvolved_presence_in_britannia_1
		}

		# Finally, queue the event up again for a year hence.
		trigger_event = {
			id = BTWK_britannia_struggle.0001
			years = 1
		}
	}
}

# Struggle phase events
# Triggered one day after new phase - figure out what the new phase is and fire the correect event
BTWK_britannia_struggle.0010 = {
	hidden = yes
	scope = none
	scope = struggle

	immediate = {
		if = {
			limit = { is_struggle_phase = struggle_britannia_phase_invasion }
			every_involved_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0011
			}
			every_interloper_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0011
			}
		}
		else_if = {
			limit = { is_struggle_phase = struggle_britannia_phase_detente }
			every_involved_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0012
			}
			every_interloper_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0012
			}
		}
		else_if = {
			limit = { is_struggle_phase = struggle_britannia_phase_consolidation }
			every_involved_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0013
			}
			every_interloper_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0013
			}
		}
		else_if = {
			limit = { is_struggle_phase = struggle_britannia_phase_unification }
			every_involved_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0014
			}
			every_interloper_ruler = {
				limit = { is_ai = no }
				trigger_event = BTWK_britannia_struggle.0014
			}
		}
	}
}

BTWK_britannia_struggle.0020 = {
	type = character_event
	content_source = dlc_BTWK
	window = fullscreen_event
	title = BTWK_britannia_struggle.0020.t
	desc = {
		first_valid = {
		    triggered_desc = {
		        trigger = {
					this = scope:conqueror
		        }
		        desc = BTWK_britannia_struggle.0020.desc.conqueror
		    }
		    desc = BTWK_britannia_struggle.0020.desc
		}
	}
	theme = war
	override_background = fp2_fullscreen_intro
	
	left_portrait = {
		character = scope:conqueror
		animation = personality_war_over_win
	}
	
	immediate = {
		if = {
			limit = { this != scope:conqueror }
			custom_tooltip = BTWK_britannia_struggle.0020.a.tt
			show_as_tooltip = { BTWK_end_britannia_struggle_effect = yes }
		}
		else = {
		}
	}

	option = {
		name = BTWK_britannia_struggle.0020.a.conqueror
		trigger = {
			this = scope:conqueror
		}
		custom_tooltip = BTWK_britannia_struggle.0020.a.tt
		BTWK_end_britannia_struggle_effect = yes
		if = {
			limit = { has_dlc_feature = legends }
			legend_seed_struggle_ending_effect = {
				ENDER = root
				STRUGGLE = britannia_struggle
			}
		}
		struggle:britannia_struggle = { end_struggle = BTWK_britannia_struggle_outsider_ending_decision }
		dynasty = {
			add_dynasty_modifier = BTWK_found_romanized_germanic_empire_britannia_modifier
		}
	}

	option = {
		name = {
			trigger = {
				OR = {
					is_britannic_culture_trigger = yes
					AND = {
						culture = { has_cultural_pillar = heritage_latin }
						primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
					}
				}
			}
			text = BTWK_britannia_struggle.0020.a.briton
		}
		name = {
			trigger = {
				is_unromanized_germanic_culture_trigger = yes
				primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
			text = BTWK_britannia_struggle.0020.a.anglo_saxon
		}
		name = {
			trigger = {
				culture = { has_cultural_pillar = heritage_latin }
				is_britannic_culture_trigger = no
				NOT = { primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia } }
			}
			text = BTWK_britannia_struggle.0020.a.roman
		}
		name = {
			trigger = {
				NOR = {
					is_britannic_culture_trigger = yes
					is_unromanized_germanic_culture_trigger = yes
					culture = { has_cultural_pillar = heritage_latin }
				}
				primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
			}
			text = BTWK_britannia_struggle.0020.a.other_in_britannia
		}
		name = {
			trigger = {
				NOR = {
					is_britannic_culture_trigger = yes
					culture = { has_cultural_pillar = heritage_latin }
					primary_title.title_capital_county.title_province = { geographical_region = btwk_world_britannia }
				}
			}
			text = BTWK_britannia_struggle.0020.a
		}
		trigger = {
			NOT = { this = scope:conqueror }
		}
	}
}

BTWK_anglo_saxon_invasions.0101 = {
	scope = none
	hidden = yes
	
	trigger = {
		exists = struggle:britannia_struggle
		BTWK_rise_of_anglo_saxon_kingdom_is_kingdom_visible_trigger = {
			KINGDOM = scope:event_target_kingdom
		}
		BTWK_rise_of_anglo_saxon_kingdom_is_kingdom_eligible_trigger = {
			KINGDOM = scope:event_target_kingdom
		}
	}
	
	immediate = {
		# Roll for chance of actually triggering the event
		random = {
			chance = {
				value = 2.5
				if = {
					limit = { year >= 540 }
					add = 2.5
				}
				if = {
					limit = { year >= 545 }
					add = 5
				}
				if = {
					limit = { year >= 550 }
					add = 10
				}
				if = {
					limit = { year >= 560 }
					add = 20
				}
				if = {
					limit = {
						scope:event_target_kingdom = title:k_mercia
						year <= 545
					}
					divide = 2
				}
				# If we already passed the check but are waiting for the historical dynasty to
				# become available, then we ignore this roll
				if = {
					limit = { 
						scope:event_target_kingdom = {							
							has_variable = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char
						}
					}
					add = 100
				}
			}
			
			# If the roll goes through, try to choose an eligible ruler
			# Always try to select the historical dynasty first
			if = {
				limit = { scope:event_target_kingdom = title:k_bernicia }
				dynasty:662 = { save_scope_as = historical_dynasty }
			}
			else_if = {
				limit = { scope:event_target_kingdom = title:k_deira }
				dynasty:663 = { save_scope_as = historical_dynasty }
			}
			else_if = {
				limit = { scope:event_target_kingdom = title:k_lindsey }
				dynasty:817 = { save_scope_as = historical_dynasty }
			}
			else_if = {
				limit = { scope:event_target_kingdom = title:k_mercia }
				dynasty:660 = { save_scope_as = historical_dynasty }
			}
			else_if = {
				limit = { scope:event_target_kingdom = title:k_wessex }
				dynasty:816 = { save_scope_as = historical_dynasty }
			}
			random_ruler = {
				limit = {
					is_landed = yes
					is_ai = yes
					BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
					BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_trigger = yes
					scope:event_target_kingdom = scope:target_kingdom
					# Prioritize historical dynasties if possible
					scope:historical_dynasty ?= dynasty
				}
				alternative_limit = {
					is_landed = yes
					is_ai = yes
					BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
					BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_excluding_temp_factors_trigger = yes
					scope:event_target_kingdom = scope:target_kingdom
					# Prioritize historical dynasties if possible
					scope:historical_dynasty ?= dynasty
				}
				save_scope_as = target_character
			}
			# Scylding adventurers are specifically chosen here for Mercia 
			# (they're not a ruler so they don't get caught in the above)
			if = {
				limit = { 
					scope:event_target_kingdom = title:k_mercia
					NOT = { exists = scope:target_character } 
				}
				title:d_laamp_scyldings.holder ?= {
					if = {
						limit = {
							is_ai = yes
							BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
							BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_excluding_temp_factors_trigger = yes
							scope:historical_dynasty ?= dynasty
						}
						save_scope_as = target_character
					}
				}
			}
			
			# If a historical dynasty character is fully eligible, fire the event
			if = {
				limit = {
					scope:target_character ?= {
						BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
						BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_trigger = yes
					}
				}
				scope:target_character = {
					trigger_event = BTWK_anglo_saxon_invasions.0110
				}
			}
			# If a historical dynasty character is eligible but busy (e.g. at war), wait up
			# to 5 years for them to become available
			else_if = {
				limit = {
					exists = scope:target_character
					NOT = { has_variable = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char }
				}
				scope:event_target_kingdom = { 
					set_variable = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char
					set_variable = {
						name = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char_timer
						years = 5
					}
				}
			}
			# If a historical dynasty character is eligible but busy and the timer is still ongoing, just wait
			else_if = {
				limit = {
					exists = scope:target_character
					has_variable = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char
					has_variable = BTWK_rise_of_anglo_saxon_kingdom_wait_for_historical_char_timer
				}
				# Do nothing
			}
			# If there are no historical dynasty characters eligible, or they are all busy and the
			# timer is up, try to look for any eligible ruler
			else = {
				random_ruler = {
					limit = {
						is_landed = yes
						is_ai = yes
						BTWK_rise_of_anglo_saxon_kingdom_is_visible_trigger = yes
						BTWK_rise_of_anglo_saxon_kingdom_is_character_valid_ai_trigger = yes
						NOT = { owns_story_of_type = BTWK_rise_of_new_anglo_saxon_kingdom_story }
						scope:event_target_kingdom = scope:target_kingdom
					}
					save_scope_as = target_character
				}
				if = {
					limit = { NOT = { exists = scope:target_character } }
					# If there are no eligible rulers, try to find a character of the historical dynasty and
					# make them an adventurer 
					scope:historical_dynasty ?= {
						random_dynasty_member = {
							limit = {
								is_adult = yes
								is_female = no
								is_landless_adventurer = yes
								is_unromanized_germanic_culture_trigger = yes
							}
							alternative_limit = {
								is_adult = yes
								is_female = no
								is_landed = no
								is_unromanized_germanic_culture_trigger = yes
							}
							save_scope_as = target_character
							scope:target_character = {
								create_landless_adventurer_title_effect = {
									REASON = flag:voluntary
									FLAVOR_CHAR = scope:target_character
								}
							}
							scope:event_target_kingdom = { save_scope_as = force_target_kingdom }
						}
					}
				}
				
				# If there are none, grab an adventurer of the right culture
				if = {
					limit = {	
						scope:event_target_kingdom = title:k_wessex
					}
					culture:old_saxon = { save_scope_as = target_culture }
				}
				else = {
					culture:angle = { save_scope_as = target_culture }
				}
				if = {
					limit = { NOT = { exists = scope:target_character } }
					random_in_global_list = {
						variable = laamps_tally
						limit = {
							title_domicile.domicile_location = { geographical_region = btwk_world_britannia }
							culture = scope:target_culture
							is_adult = yes
							is_female = no
							NOR = {
								any_close_or_extended_family_member = {
									highest_held_title_tier >= tier_county
								}
								any_spouse = {
									highest_held_title_tier >= tier_county
								}
							}
						}
						save_scope_as = target_character
						scope:event_target_kingdom = { save_scope_as = force_target_kingdom }
					}
				}
				
				# If there are still none, just spawn a new adventurer
				if = {
					limit = { NOT = { exists = scope:target_character } }
					create_character = {
						template = pool_repopulate_martial
						gender_female_chance = 0
						culture = scope:target_culture
						faith = faith:anglo_saxon_pagan
						location = scope:event_target_kingdom.title_capital_county.title_province
						save_scope_as = target_character
					}
					scope:target_character = {
						create_landless_adventurer_title_effect = {
							REASON = flag:voluntary
							FLAVOR_CHAR = scope:target_character
						}
					}
					scope:event_target_kingdom = { save_scope_as = force_target_kingdom }
				}
				
				# Then fire the event for the new adventurer
				scope:target_character = {
					trigger_event = BTWK_anglo_saxon_invasions.0110
				}
			}
		}
	}
}

BTWK_anglo_saxon_invasions.9000 = {
	hidden = yes
	content_source = dlc_BTWK
	
	trigger = {
		struggle:britannia_struggle ?= { has_struggle_phase_parameter = anglo_saxon_controlled_counties_can_become_tribal }
		scope:target_county = {
			title_province = {
				OR = {
					has_holding_type = castle_holding
					has_holding_type = city_holding
				}
			}
			holder = {
				is_unromanized_germanic_culture_trigger = yes
				government_has_flag = government_is_tribal
				top_liege = { government_has_flag = government_is_tribal }
			}
		}
	}

	immediate = {
		scope:target_county = {
			random = {
				chance = {
					value = 20
					add = {
						value = this.var:BTWK_britannia_county_tribal_control
						subtract = 6		# First roll occurs when the var is 6
						multiply = 2.5
					}
					if = {
						limit = { 
							NOR = {
								any_neighboring_county = { title_province = { has_holding_type = tribal_holding } }
								holder.top_liege = {
									owns_story_of_type = BTWK_rise_of_new_anglo_saxon_kingdom_story
								}
							}
						}
						divide = 4
					}
					if = {
						limit = { is_unromanized_germanic_culture_trigger = yes }
						multiply = 2
					}
					if = {
						limit = {
							holder = {
								any_owned_story = {
									story_type = BTWK_rise_of_new_anglo_saxon_kingdom_story
									this.var:target_kingdom = title:k_lindsey
								}
							}
						}
						divide = 3
					}
					if = {
						limit = { holder.top_liege = { is_unromanized_germanic_culture_trigger = no } }
						divide = 2
					}
				}
				holder = { trigger_event = BTWK_anglo_saxon_invasions.9001 }
			}
		}
	}
}

		faith = { save_scope_as = new_faith }

		name = BTWK_anglo_saxon_invasions.9003.a

			set_county_culture = holder.culture
			set_county_faith = root.faith
		}
		# Send toasts to lieges
		hidden_effect = {
			every_liege_or_above = { 
				send_interface_toast = {
					type = msg_culture_conversion
					title = culture_shifts.9999.t
					custom_tooltip = culture_shifts.9999.desc
					right_icon = scope:current_county
				}
				if = {
					limit = { scope:old_faith != scope:new_faith }
					switch = {
						trigger = faith
						scope:current_county.faith = { # Good if same as new faith
							send_interface_message = {
								type = msg_council_conversion_good
								title = BTWK_anglo_saxon_invasions.9003.toast.t
								desc = BTWK_anglo_saxon_invasions.9003.toast.desc
								right_icon = scope:current_county
							}
						}
						scope:old_faith = {
							send_interface_message = { # Bad if same faith as county
								type = msg_council_conversion_bad
								title = BTWK_anglo_saxon_invasions.9003.toast.t
								desc = BTWK_anglo_saxon_invasions.9003.toast.desc
								right_icon = scope:current_county
							}
						}
						fallback = {
							send_interface_message = { # Neutral otherwise
								type = msg_council_conversion_neutral
								title = BTWK_anglo_saxon_invasions.9003.toast.t
								desc = BTWK_anglo_saxon_invasions.9003.toast.desc
								right_icon = scope:current_county
							}
						}
					}
				}
			}
		}
	}
