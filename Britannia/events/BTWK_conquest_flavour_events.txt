# Master controller event
conquest_flavour_anglosaxon.0001 = {
	scope = none
	content_source = dlc_BTWK
	hidden = yes
	
	immediate = {
		title:e_aengland.holder ?= { 
			save_scope_as = conqueror
			if = {
				limit = {
					title:e_aengland = { NOT = { has_variable = aengland_conquered } }
					completely_controls_region = btwk_custom_aengland
				}
				title:e_aengland = { set_variable = aengland_conquered }
				trigger_event = { id = conquest_flavour_anglosaxon.0002 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_anglosaxon.0002 }
				}
			}
			if = {
				limit = {
					title:e_aengland = { NOT = { has_variable = wales_conquered } }
					completely_controls_region = btwk_custom_wales
				}
				title:e_aengland = { set_variable = wales_conquered }
				trigger_event = { id = conquest_flavour_anglosaxon.0003 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_anglosaxon.0003 }
				}
			}
			if = {
				limit = {
					title:e_aengland = { NOT = { has_variable = britannia_conquered } }
					completely_controls_region = btwk_world_britannia
				}
				title:e_aengland = { set_variable = britannia_conquered }
				trigger_event = { id = conquest_flavour_anglosaxon.0004 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_anglosaxon.0004 }
				}
			}
		}
	}
}

# Unification of Aengland
conquest_flavour_anglosaxon.0002 = {
	type = character_event
	content_source = dlc_BTWK
	window = big_event_window
	title = conquest_flavour_anglosaxon.0002.t
	desc = conquest_flavour_anglosaxon.0002.desc
	theme = war
	override_background = terrain

		character = scope:conqueror
		animation = personality_war_over_win
	}

	option = {
		name = {
			trigger = {

			trigger = { 
				culture = {

						has_cultural_pillar = heritage_brythonic

# Master controller event
conquest_flavour_briton.0001 = {
	scope = none
	content_source = dlc_BTWK
	hidden = yes
	
	immediate = {
		title:c_londinium.holder.top_liege = {
			save_scope_as = conqueror
			if = {
				limit = {
					title:e_britannia = { NOT = { has_variable = anglo_saxons_defeated_south } }
					NOT = {
						any_county_in_region = {
							region = btwk_world_britannia_south
							holder.top_liege = {
								OR = {
									is_unromanized_germanic_culture_trigger = yes
									primary_title.empire = title:e_aengland
								}
							}
						}
					}
					# TODO Need to wait until all relevant Anglo-Saxon invasions have been triggered and defeated
				}
				title:e_britannia = { set_variable = anglo_saxons_defeated_south }
				# Done here to make sure that no weirdness happens if both events are triggered at once
				if = {
					limit = { title:e_britannia = { has_variable = anglo_saxons_defeated_north } }
					title:e_britannia = { set_variable = anglo_saxons_defeated_all }
				}
				trigger_event = { id = conquest_flavour_briton.0002 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_briton.0002 }
				}
			}
		}
		title:c_eburacum.holder.top_liege = {
			if = {
				limit = {
					title:e_britannia = { NOT = { has_variable = anglo_saxons_defeated_north } }
					NOT = {
						any_county_in_region = {
							region = btwk_world_britannia_hen_ogledd
							holder.top_liege = {
								OR = {
									is_unromanized_germanic_culture_trigger = yes
									primary_title.empire = title:e_aengland
								}
							}
						}
					}
					# TODO Need to wait until all relevant Anglo-Saxon invasions have been triggered and defeated
				}
				title:e_britannia = { set_variable = anglo_saxons_defeated_north }
				if = {
					limit = { title:e_britannia = { has_variable = anglo_saxons_defeated_south } }
					title:e_britannia = { set_variable = anglo_saxons_defeated_all }
				}
				trigger_event = { id = conquest_flavour_briton.0003 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_briton.0003 }
				}
			}
		}
		title:e_britannia.holder ?= {
			save_scope_as = conqueror
			if = {
				limit = {
					title:e_britannia = { NOT = { has_variable = britannia_conquered } }
					completely_controls_region = btwk_world_britannia
				}
				title:e_britannia = { set_variable = britannia_conquered }
				trigger_event = { id = conquest_flavour_briton.0004 }
				every_player = {
					limit = { NOT = { this = PREV } }
					trigger_event = { id = conquest_flavour_briton.0004 }
				}
			}
		}
	}
}
