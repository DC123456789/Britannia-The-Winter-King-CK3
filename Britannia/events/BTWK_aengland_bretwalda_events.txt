# BTWK_aengland_bretwalda.0001 - King becomes Bretwalda

# King becomes Bretwalda
BTWK_aengland_bretwalda.0001 = { 
	type = character_event
	content_source = dlc_BTWK
	window = big_event_window
	title = BTWK_aengland_bretwalda.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:new_bretwalda = root }
				desc = BTWK_aengland_bretwalda.0001.desc.bretwalda
			}
			desc = BTWK_aengland_bretwalda.0001.desc.other
		}
	}
	theme = war
	override_background = { reference = throne_room_scope }

	left_portrait = {
		character = scope:new_bretwalda
		animation = war_over_win
	}
	
	immediate = {
		scope:new_bretwalda = { 
			save_scope_as = background_throne_room_scope 
			capital_barony = { save_scope_as = new_bretwalda_capital_barony }
		}
		if = {
			limit = { scope:new_bretwalda = root }
			title:e_aengland = {
				set_variable = {
					name = aengland_primary_kingdom
					value = scope:old_primary_title
				}
				set_capital_county = scope:old_primary_title.title_capital_county
			}
		}
		else = {
			show_as_tooltip = {
				scope:new_bretwalda = { get_title = title:e_aengland }
			}
		}
	}

	option = {
		name = BTWK_aengland_bretwalda.0001.a
		trigger = { scope:new_bretwalda = root }
		add_prestige = 200
		get_title = title:e_aengland
		save_scope_value_as = {
			name = hegemony_score_starting_boost
			value = 30
		}
		every_held_title = {
			limit = { has_variable = inherited_aengland_bretwalda_hegemony_score }
			if = {
				limit = { this.var:inherited_aengland_bretwalda_hegemony_score > scope:hegemony_score_starting_boost }
				save_scope_value_as = {
					name = hegemony_score_starting_boost
					value = this.var:inherited_aengland_bretwalda_hegemony_score
				}
				save_scope_as = inherited_hegemony_score
			}
		}
		if = {
			limit = { exists = scope:inherited_hegemony_score }
			custom_tooltip = {
				text = BTWK_aengland_bretwalda.0001.a.gain_hegemony_score_inherited.tt
				change_variable = {
					name = aengland_bretwalda_hegemony_score
					add = scope:hegemony_score_starting_boost
				}
			}
		}
		else = {
			custom_tooltip = {
				text = BTWK_aengland_bretwalda.0001.a.gain_hegemony_score.tt
				change_variable = {
					name = aengland_bretwalda_hegemony_score
					add = scope:hegemony_score_starting_boost
				}
			}
		}
		every_kingdom = {
			limit = {
				has_variable = inherited_aengland_bretwalda_hegemony_score
			}
			remove_variable = inherited_aengland_bretwalda_hegemony_score
		}
		custom_tooltip = BTWK_aengland_bretwalda.0001.a.tt.1
		custom_tooltip = BTWK_aengland_bretwalda.0001.a.tt.2
		dynasty = { add_dynasty_prestige = major_dynasty_prestige_gain }
		hidden_effect = {
			struggle:britannia_struggle ?= {
				if = {
					limit = { phase_has_catalyst = catalyst_become_bretwalda_aengland }
					activate_struggle_catalyst = {
						catalyst = catalyst_become_bretwalda_aengland
						character = root
					}
				}
			}
		}
	}

	option = {
		name = {
			trigger = {
				is_unromanized_germanic_culture_trigger = yes
				any_held_title = {
					tier = root.primary_title.tier
					title_capital_county.title_province = { geographical_region = btwk_world_britannia }
				}
				top_liege = {
					OR = {
						this = scope:new_bretwalda
						suzerain = scope:new_bretwalda
					}
				}
			}
			text = BTWK_aengland_bretwalda.0001.a.anglo_saxon_vassal
		}
		name = {
			trigger = {
				is_unromanized_germanic_culture_trigger = yes
				any_held_title = {
					tier = root.primary_title.tier
					title_capital_county.title_province = { geographical_region = btwk_world_britannia }
				}
				is_independent_ruler = yes
				NOT = {
					is_tributary = yes
					suzerain = scope:new_bretwalda
				}
			}
			text = BTWK_aengland_bretwalda.0001.a.anglo_saxon_other_king
		}
		name = {
			trigger = {
				is_unromanized_germanic_culture_trigger = yes
				any_held_title = {
					tier = root.primary_title.tier
					title_capital_county.title_province = { geographical_region = btwk_world_britannia }
				}
				NAND = {
					is_independent_ruler = yes
					NOT = {
						is_tributary = yes
						suzerain = scope:new_bretwalda
					}
				}
			}
			text = BTWK_aengland_bretwalda.0001.a.anglo_saxon
		}
		name = {
			trigger = {
				is_britannic_culture_trigger = yes
			}
			text = BTWK_aengland_bretwalda.0001.a.briton_roman
		}
		name = {
			trigger = {
				NOR = {
					is_britannic_culture_trigger = yes
					AND = {
						is_unromanized_germanic_culture_trigger = yes
						any_held_title = {
							tier = root.primary_title.tier
							title_capital_county.title_province = { geographical_region = btwk_world_britannia }
						}
					}
				}
			}
			text = BTWK_aengland_bretwalda.0001.a.other
		}
		trigger = { scope:new_bretwalda != root }
	}
}

			}

	}
}
