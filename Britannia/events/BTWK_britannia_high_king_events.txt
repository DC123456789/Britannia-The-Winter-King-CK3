############################################################################
# BTWK_britannia_high_king.0001 - Claim made
# BTWK_britannia_high_king.0002 - Other kings notified of claim made
# BTWK_britannia_high_king.0003 - Bounce event to trigger asking other kings 
# BTWK_britannia_high_king.0004 - Other kings decide on stance
# BTWK_britannia_high_king.0005 - Notify claimant of responses
# BTWK_britannia_high_king.0006 - Claimant decides whether to go to war or give up their claim
# BTWK_britannia_high_king.0007 - Notify other kings of claimant's decision to proceed or not
# BTWK_britannia_high_king.0008 - Claimant won war / no opponents, checks that you are still eligible
# BTWK_britannia_high_king.0009 - Claimant becomes High King (Claimant POV)
# BTWK_britannia_high_king.0010 - Claimant becomes High King (Other POV)
# BTWK_britannia_high_king.0011 - Claimant white peaced
# BTWK_britannia_high_king.0012 - Claimant defeated
# BTWK_britannia_high_king.0013 - Claim attempt ends inconclusively due to claimant dying or losing their throne

# High Kingship claimed, awaiting responses

# Trigger actual High King support request event for other kings
BTWK_britannia_high_king.0003 = {
	hidden = yes
	content_source = dlc_BTWK
	
	trigger = {
		title:e_britannia.var:britannia_high_kingship_claimant ?= ROOT
	}

	immediate = {
		every_ruler = {
			limit = { 
				BTWK_britannia_high_kingship_is_elector_trigger = yes
				NOT = { this = ROOT }
			}
			trigger_event = BTWK_britannia_high_king.0004
		}
	}
}

# Other kings receive request for support, decide stance
BTWK_britannia_high_king.0004 = {
	type = letter_event
	opening = BTWK_britannia_high_king.0004.opening
	desc = BTWK_britannia_high_king.0004.desc
	
	sender = {
		character = scope:claimant
		animation = personality_rational
	}
	
	immediate = {
		save_scope_as = other_king
	}

	option = {
		name = BTWK_britannia_high_king.0004.a
		if = {
			limit = {
				OR = {
					is_tributary = no
					suzerain ?= scope:claimant
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt
		}
		else_if = {
			limit = { suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt.other_tributary
		}
		if = {
			limit = {
				any_tributary = {
					BTWK_britannia_high_kingship_is_elector_trigger = yes
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt.suzerain_warning
		}
		set_variable = HK_britannia_supporter
		add_to_global_variable_list = {
			name = HK_britannia_supporter_list
			target = THIS
		}
		scope:claimant = { trigger_event = BTWK_britannia_high_king.0005 }
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_supporter_threshold 
			}
		}
	}
	option = {
		name = BTWK_britannia_high_king.0004.b
		if = {
			limit = {
				OR = {
					is_tributary = no
					suzerain ?= scope:claimant
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt
		}
		else_if = {
			limit = { suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt.other_tributary
		}
		if = {
			limit = {
				any_tributary = {
					BTWK_britannia_high_kingship_is_elector_trigger = yes
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt.suzerain_warning
		}
		set_variable = HK_britannia_opponent
		add_to_global_variable_list = {
			name = HK_britannia_opponent_list
			target = THIS
		}
		scope:claimant = { trigger_event = BTWK_britannia_high_king.0005 }
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant < BTWK_britannia_high_king_scope_opponent_threshold 
			}
		}
	}
	option = {
		name = BTWK_britannia_high_king.0004.c
		if = {
			limit = { scope:other_king = { is_tributary = no } }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt
		}
		else_if = {
			limit = { scope:other_king.suzerain ?= root }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.claimant_tributary
		}
		else_if = {
			limit = { scope:other_king.suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.other_tributary
		}
		scope:claimant = { 
			if = {
				limit = { NOT = { has_variable = received_BTWK_britannia_high_king_0005 } }
				trigger_event = BTWK_britannia_high_king.0005 
				set_variable = received_BTWK_britannia_high_king_0005
			}
		}
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant < BTWK_britannia_high_king_scope_supporter_threshold 
				BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_opponent_threshold 
			}
		}
	}
}

# All responses recieved, decide what to do now
BTWK_britannia_high_king.0006 = { 
	type = character_event
	content_source = dlc_BTWK
	title = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } 
					title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try }
				}
				desc = BTWK_britannia_high_king.0006.t.second_try_no_opponents
			}
			triggered_desc = {
				trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
				desc = BTWK_britannia_high_king.0006.t.no_opponents
			}
			triggered_desc = {
				trigger = { title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try } }
				desc = BTWK_britannia_high_king.0006.t.second_try
			}
			desc = BTWK_britannia_high_king.0006.t
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } 
					title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try }
				}
				desc = BTWK_britannia_high_king.0006.desc.second_try_no_opponents
			}
			triggered_desc = {
				trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
				desc = BTWK_britannia_high_king.0006.desc.no_opponents
			}
			triggered_desc = {
				trigger = { title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try } }
				desc = BTWK_britannia_high_king.0006.desc.second_try
			}
			desc = BTWK_britannia_high_king.0006.desc
		}
	}
	theme = war
	override_background = { reference = throne_room }

	left_portrait = {
		character = root
		animation = war_over_win
	}

	right_portrait = {
		character = scope:HK_britannia_opponent_leader
		animation = war_defender
		trigger = { exists = scope:HK_britannia_opponent_leader }
	}
	
	immediate = {
		ordered_independent_ruler = {
			limit = { has_variable = HK_britannia_opponent }
			order_by = military_power
			save_scope_as = HK_britannia_opponent_leader
		}
		title:e_britannia = { set_variable = britannia_high_kingship_stances_decided }
		remove_variable = received_BTWK_britannia_high_king_0005
	}

	option = {		# To war!
		name = BTWK_britannia_high_king.0006.a
		trigger = { any_independent_ruler = { has_variable = HK_britannia_opponent } }
		# Declare war on your strongest opponent
		hidden_effect = {
			if = {
				limit = { has_truce = scope:HK_britannia_opponent_leader }
				cancel_truce_one_way = scope:HK_britannia_opponent_leader
			}
		}
		start_war = {
			cb = BTWK_dispute_britannia_high_kingship_cb
			target = scope:HK_britannia_opponent_leader
		}
		title:e_britannia = { set_variable = britannia_high_kingship_dispute_war }
		random_character_war = {
			limit = {
				primary_attacker = root
				primary_defender = scope:HK_britannia_opponent_leader
				using_cb = BTWK_dispute_britannia_high_kingship_cb
			}
			save_scope_as = HK_britannia_war
		}
		custom_tooltip = BTWK_britannia_high_king.0006.a.tt.1
		if = {
			limit = { NOT = { title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try } } }
			custom_tooltip = BTWK_britannia_high_king.0006.a.tt.2
			custom_tooltip = BTWK_britannia_high_king.0006.a.tt.3
		}
		# All supporters join your side
		every_independent_ruler = {
			limit = { has_variable = HK_britannia_supporter }
			scope:HK_britannia_war ?= { add_attacker = prev }
		}
		# All opponents join the enemy
		every_independent_ruler = {
			limit = { 
				has_variable = HK_britannia_opponent
				NOT = { this = scope:HK_britannia_opponent_leader }
			}
			scope:HK_britannia_war ?= { add_defender = prev }
		}
		# Notify other players
		save_scope_as = claimant
		every_player = {
			limit = { 
				NOT = { this = root }
				BTWK_britannia_high_kingship_is_relevant_trigger = yes
			}
			trigger_event = BTWK_britannia_high_king.0007
		}
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try }
			}
			modifier = {
				add = 100
				BTWK_britannia_high_king_dispute_war_military_power_ratio > 1.2		# Only if they think they can win
			}
		}
	}
	option = {		# Back down
		name = BTWK_britannia_high_king.0006.b
		trigger = { any_independent_ruler = { has_variable = HK_britannia_opponent } }
		add_prestige = -200
		custom_tooltip = BTWK_britannia_high_king_abandon_claim_claimant.tt
		add_character_modifier = {
			modifier = BTWK_recently_claimed_high_kingship_modifier
			years = 5
		}
		title:e_britannia = { remove_variable = britannia_high_kingship_dispute_war }
		every_owned_story = {
			limit = { story_type = BTWK_britannia_claim_high_kingship_story }
			set_variable = claimant_backed_down
			end_story = yes
		}
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_dispute_war_military_power_ratio < 1.1		# If they don't think they can win
				NOT = { title:e_britannia = { has_variable = britannia_high_kingship_dispute_war_second_try } }
			}
		}
	}
	option = {		# No opponents, automatically win
		name = BTWK_britannia_high_king.0006.c
		trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
		custom_tooltip = BTWK_britannia_high_king.0006.c.tt
		trigger_event = BTWK_britannia_high_king.0008
	}
}
