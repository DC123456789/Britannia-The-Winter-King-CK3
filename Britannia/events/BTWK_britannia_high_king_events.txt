# High Kingship claimed, awaiting responses
BTWK_britannia_high_king.0001 = { 
	type = character_event
	content_source = dlc_BTWK

# Trigger actual High King support request event for other kings
BTWK_britannia_high_king.0003 = {
	hidden = yes
	content_source = dlc_BTWK
	
	trigger = {
		title:e_britannia.var:britannia_high_kingship_claimant ?= ROOT
	}

	immediate = {
		every_ruler = {
			limit = { 
				BTWK_britannia_high_kingship_is_elector_trigger = yes
				NOT = { this = ROOT }
			}
			trigger_event = BTWK_britannia_high_king.0004
		}
	}
}

# Other kings receive request for support, decide stance
BTWK_britannia_high_king.0004 = {
	type = letter_event
	opening = BTWK_britannia_high_king.0004.opening
	desc = BTWK_britannia_high_king.0004.desc
	
	sender = {
		character = scope:claimant
		animation = personality_rational
	}
	
	immediate = {
		save_scope_as = other_king
	}

	option = {
		name = BTWK_britannia_high_king.0004.a
		if = {
			limit = {
				OR = {
					is_tributary = no
					suzerain ?= scope:claimant
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt
		}
		else_if = {
			limit = { suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.a.tt.other_tributary
		}
		set_variable = HK_britannia_supporter
		scope:claimant = { trigger_event = BTWK_britannia_high_king.0005 }
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_supporter_threshold 
			}
		}
	}
	option = {
		name = BTWK_britannia_high_king.0004.b
		if = {
			limit = {
				OR = {
					is_tributary = no
					suzerain ?= scope:claimant
				}
			}
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt
		}
		else_if = {
			limit = { suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.b.tt.other_tributary
		}
		set_variable = HK_britannia_opponent
		scope:claimant = { trigger_event = BTWK_britannia_high_king.0005 }
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant < BTWK_britannia_high_king_scope_opponent_threshold 
			}
		}
	}
	option = {
		name = BTWK_britannia_high_king.0004.c
		if = {
			limit = { scope:other_king = { is_tributary = no } }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt
		}
		else_if = {
			limit = { scope:other_king.suzerain ?= root }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.claimant_tributary
		}
		else_if = {
			limit = { scope:other_king.suzerain ?= { BTWK_britannia_high_kingship_is_elector_trigger = yes } }
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.briton_tributary
		}
		else = {
			custom_tooltip = BTWK_britannia_high_king.0004.c.tt.other_tributary
		}
		scope:claimant = { trigger_event = BTWK_britannia_high_king.0005 }
		
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				BTWK_britannia_high_king_scope_support_for_claimant < BTWK_britannia_high_king_scope_supporter_threshold 
				BTWK_britannia_high_king_scope_support_for_claimant >= BTWK_britannia_high_king_scope_opponent_threshold 
			}
		}
	}
}

# Claimant receives responses from other kings

				desc = BTWK_britannia_high_king.0005.opening.supporter
			}
			triggered_desc = {
				trigger = { scope:other_king = { has_variable = HK_britannia_opponent } }
				desc = BTWK_britannia_high_king.0005.opening.opponent
			}
			desc = BTWK_britannia_high_king.0005.opening.neutral
		}
	}
	desc = {
		first_valid = {

				trigger = { 
					scope:other_king = {

		trigger = { scope:other_king = { has_variable = HK_britannia_opponent } }
		if = {
			limit = {
				OR = {

			}

# All responses recieved, decide what to do now
BTWK_britannia_high_king.0006 = { 
	type = character_event
	content_source = dlc_BTWK
	title = {
		first_valid = {
			triggered_desc = {
				trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
				desc = BTWK_britannia_high_king.0006.t.no_opponents
			}
			desc = BTWK_britannia_high_king.0006.t
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
				desc = BTWK_britannia_high_king.0006.desc.no_opponents
			}
			desc = BTWK_britannia_high_king.0006.desc
		}
	}
	theme = war
	override_background = { reference = throne_room }

	left_portrait = {
		character = root
		animation = war_over_win
	}
	
	immediate = {
		ordered_independent_ruler = {
			limit = { has_variable = HK_britannia_opponent }
			order_by = military_power
			save_scope_as = HK_britannia_opponent_leader
		}
	}

	option = {		# To war!
		name = BTWK_britannia_high_king.0006.a
		trigger = { any_independent_ruler = { has_variable = HK_britannia_opponent } }
		custom_tooltip = BTWK_britannia_high_king.0006.a.tt.1
		custom_tooltip = BTWK_britannia_high_king.0006.a.tt.2
		# TODO Declare war with custom cb on your strongest opponent
		# All supporters join your side
		# All opponents join the enemy
	}
	option = {		# Back down
		name = BTWK_britannia_high_king.0006.b
		trigger = { any_independent_ruler = { has_variable = HK_britannia_opponent } }
		add_prestige = -200
		custom_tooltip = BTWK_britannia_high_king.0006.b.tt
		add_character_modifier = {
			modifier = BTWK_recently_claimed_high_kingship_modifier
			years = 5
		}
		every_owned_story = {
			limit = { story_type = BTWK_britannia_claim_high_kingship_story }
			end_story = yes
		}
	}
	option = {		# No opponents, automatically win
		name = BTWK_britannia_high_king.0006.c
		trigger = { NOT = { any_independent_ruler = { has_variable = HK_britannia_opponent } } }
		custom_tooltip = BTWK_britannia_high_king.0006.c.tt
		trigger_event = BTWK_britannia_high_king.0008
	}
}
