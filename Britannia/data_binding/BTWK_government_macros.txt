macro = {
	description = "Is this a BTWK feudal government"
	definition = "IsBTWKFeudalGovernment(TargetGovernment)"
	replace_with = "Or(TargetGovernment.IsType('sub_roman_brythonic_government'), Or(TargetGovernment.IsType('sub_roman_no_dlc_government'), TargetGovernment.IsType('germanic_no_dlc_government')))"
}

macro = {
	description = "Is this a BTWK administrative government"
	definition = "IsBTWKAdminGovernment(TargetGovernment)"
	replace_with = "Or(TargetGovernment.IsType('sub_roman_government'), TargetGovernment.IsType('germanic_government'))"
}
