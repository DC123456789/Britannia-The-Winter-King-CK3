macro = {
	description = "Does the bookmark character have an Anglo-Saxon culture"
	definition = "BookmarkCharacterIsAngloSaxon(BookmarkCharacter)"
	replace_with = "Or(Or(EqualTo_string(Localize(BookmarkCharacter.GetCultureName), Localize('jute')), EqualTo_string(Localize(BookmarkCharacter.GetCultureName), Localize('angle'))), Or(EqualTo_string(Localize(BookmarkCharacter.GetCultureName), Localize('old_saxon')), EqualTo_string(Localize(BookmarkCharacter.GetCultureName), Localize('anglo_saxon'))))"
}
